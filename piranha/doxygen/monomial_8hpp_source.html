404  template <typename U = T, linarg_enabler<U> = 0>
405  std::string linear_argument(const symbol_set &args) const
406  {
407  if (!is_compatible(args)) {
408  piranha_throw(std::invalid_argument,"invalid size of arguments set");
409  }
410  typedef typename base::size_type size_type;
411  const size_type size = this->size();
412  size_type n_linear = 0u, candidate = 0u;
413  for (size_type i = 0u; i < size; ++i) {
414  integer tmp;
415  try {
416  tmp = safe_cast<integer>((*this)[i]);
417  } catch (const std::invalid_argument &) {
418  piranha_throw(std::invalid_argument,"exponent is not an integer");
419  }
420  if (tmp == 0) {
421  continue;
422  }
423  if (tmp != 1) {
424  piranha_throw(std::invalid_argument,"exponent is not unitary");
425  }
426  candidate = i;
427  ++n_linear;
428  }
429  if (n_linear != 1u) {
430  piranha_throw(std::invalid_argument,"monomial is not linear");
431  }
432  return args[candidate].get_name();
433  }

497  template <typename U = T, partial_enabler<U> = 0>
498  std::pair<U,monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
499  {
500  if (!is_compatible(args)) {
501  piranha_throw(std::invalid_argument,"invalid size of arguments set");
502  }
503  // Cannot take derivative wrt more than one variable, and the position of that variable
504  // must be compatible with the monomial.
505  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
506  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
507  }
508  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
509  // variable with zero exponent.
510  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
511  // there is a valid position and hence the size must be not zero.
512  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
513  return std::make_pair(U(0),monomial(args));
514  }
515  // Copy of the original monomial.
516  monomial m(*this);
517  auto m_b = m.begin();
518  // Original exponent.
519  U v(m_b[*p.begin()]);
520  // Decrement the exponent in the monomial.
521  ip_dec(m_b[*p.begin()]);
522  return std::make_pair(std::move(v),std::move(m));
523  }

546  std::pair<T,monomial> integrate(const symbol &s, const symbol_set &args) const
547  {
548  typedef typename base::size_type size_type;
549  if (!is_compatible(args)) {
550  piranha_throw(std::invalid_argument,"invalid size of arguments set");
551  }
552  monomial retval;
553  T expo(0), one(1);
554  in_place_adder<T> adder;
555  for (size_type i = 0u; i < args.size(); ++i) {
556  if (math::is_zero(expo) && s < args[i]) {
557  // If we went past the position of s in args and still we
558  // have not performed the integration, it means that we need to add
559  // a new exponent.
560  retval.push_back(one);
561  expo = one;
562  }
563  retval.push_back((*this)[i]);
564  if (args[i] == s) {
565  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
566  // we are never going to land here as args[i] is at this point never going to be s.
567  adder(retval[i],one);
568  if (math::is_zero(retval[i])) {
569  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
570  }
571  expo = retval[i];
572  }
573  }
574  // If expo is still zero, it means we need to add a new exponent at the end.
575  if (math::is_zero(expo)) {
576  retval.push_back(one);
577  expo = one;
578  }
579  return std::make_pair(std::move(expo),std::move(retval));
580  }

595  {
596  if (unlikely(args.size() != this->size())) {
597  piranha_throw(std::invalid_argument,"invalid size of arguments set");
598  }
599  const T zero(0), one(1);

602  if ((*this)[i] != zero) {
603  // If we are going to print a symbol, and something has been printed before,
604  // then we are going to place the multiplication sign.
605  if (!empty_output) {
606  os << '*';
607  }

609  empty_output = false;

611  os << "**" << detail::prepare_for_print((*this)[i]);

629  void print_tex(std::ostream &os, const symbol_set &args) const
630  {
631  if (unlikely(args.size() != this->size())) {
632  piranha_throw(std::invalid_argument,"invalid size of arguments set");
633  }
634  std::ostringstream oss_num, oss_den, *cur_oss;
635  const T zero(0), one(1);
636  T cur_value;
637  for (typename base::size_type i = 0u; i < this->size(); ++i) {
638  cur_value = (*this)[i];
639  if (cur_value != zero) {
640  cur_oss = (cur_value > zero) ? std::addressof(oss_num) : (math::negate(cur_value),std::addressof(oss_den));
641  (*cur_oss) << "{" << args[i].get_name() << "}";
642  if (cur_value != one) {
643  (*cur_oss) << "^{" << detail::prepare_for_print(cur_value) << "}";
644  }
645  }
646  }
647  const std::string num_str = oss_num.str(), den_str = oss_den.str();
648  if (!num_str.empty() && !den_str.empty()) {
649  os << "\\frac{" << num_str << "}{" << den_str << "}";
650  } else if (!num_str.empty() && den_str.empty()) {
651  os << num_str;
652  } else if (num_str.empty() && !den_str.empty()) {
653  os << "\\frac{1}{" << den_str << "}";
654  }

681  template <typename U>
682  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
683  {
684  using return_type = eval_type<U>;
685  using size_type = typename base::size_type;
686  // NOTE: the positions map must have the same number of elements as this, and it must
687  // be made of consecutive positions [0,size - 1].
688  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
689  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
690  }
691  // Args must be compatible with this.
692  if (unlikely(args.size() != this->size())) {
693  piranha_throw(std::invalid_argument,"invalid size of arguments set");
694  }
695  return_type retval(1);
696  auto it = pmap.begin();
697  for (size_type i = 0u; i < this->size(); ++i, ++it) {
698  piranha_assert(it != pmap.end() && it->first == i);
699  retval *= math::pow(it->second,(*this)[i]);
700  }
701  piranha_assert(it == pmap.end());
702  return retval;
703  }

834  template <typename Cf, typename U = monomial, multiply_enabler<Cf,U> = 0>
835  static void multiply(std::array<term<Cf,monomial>,multiply_arity> &res, const term<Cf,monomial> &t1,
836  const term<Cf,monomial> &t2, const symbol_set &args)

135  // Canonical term: the first nonzero element is positive and all the gcd of all elements is 1 or -1.
136  // NOTE: this also includes the check for all zero elements, as gcd(0,0,...,0) = 0.
137  static bool term_is_canonical(const p_type &p)
138  {
139  bool first_nonzero_found = false;
140  value_type cd(0);
141  for (const auto &n: p.v) {
142  if (!first_nonzero_found && !math::is_zero(n)) {
143  if (n < 0) {
144  return false;
145  }
146  first_nonzero_found = true;
147  }
148  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
149  cd = detail::gcd(cd,n);
150  }
151  if (cd != 1 && cd != -1) {
152  return false;
153  }
154  return true;
155  }

171  {

195  // Insertion machinery.
196  template <typename Term>
197  void insertion_impl(Term &&term)
198  {
199  // Handle the case of a table with no buckets.
200  if (unlikely(!m_container.bucket_count())) {
201  m_container._increase_size();
202  }
203  // Try to locate the term.
204  auto bucket_idx = m_container._bucket(term);
205  const auto it = m_container._find(term,bucket_idx);
206  if (it == m_container.end()) {
207  // New term.
208  if (unlikely(m_container.size() == std::numeric_limits<size_type>::max())) {
209  piranha_throw(std::overflow_error,"maximum number of elements reached");
210  }
211  // Term is new. Handle the case in which we need to rehash because of load factor.
212  if (unlikely(static_cast<double>(m_container.size() + size_type(1u)) / static_cast<double>(m_container.bucket_count()) >
213  m_container.max_load_factor()))
214  {
215  m_container._increase_size();
216  // We need a new bucket index in case of a rehash.
217  bucket_idx = m_container._bucket(term);
218  }
219  // Actually perform the insertion and finish by updating the size.
220  m_container._unique_insert(std::forward<Term>(term),bucket_idx);
221  m_container._update_size(m_container.size() + size_type(1u));
222  } else {
223  // Existing term - update the exponent.
224  update_exponent(it->e,term.e);
225  }
226  }
227  template <typename U = T, typename std::enable_if<std::is_integral<U>::value,int>::type = 0>
228  static void update_exponent(value_type &a, const value_type &b)
229  {
230  // Check the range when the exponent is an integral type.
231  piranha_assert(a > 0);
232  piranha_assert(b > 0);
233  // NOTE: this is safe as we require b to be a positive value.
234  if (unlikely(a > std::numeric_limits<value_type>::max() - b)) {
235  piranha_throw(std::invalid_argument,"overflow in the computation of the exponent of a divisor term");
236  }
237  a = static_cast<value_type>(a + b);
238  }

257  template <class Archive>
258  void load(Archive &ar, unsigned int)
259  {
260  divisor new_d;
261  try {
262  // NOTE: here we could throw either because the archive is garbage (Boost.Serialization throwing
263  // in this case) or because the loaded terms are invalid. In the second case we must make sure
264  // to destroy the content of new_d before exiting, otherwise the dtor of the divisor will
265  // hit assertion failures in debug mode. We need the second clear() in the catch block

270  // Run the destruction checks, if they fail throw.
271  if (unlikely(!new_d.destruction_checks())) {
272  new_d.clear();
273  piranha_throw(std::invalid_argument,"error during the deserialization of a divisor, the loaded data "
274  "is invalid");
275  }
276  } catch (...) {
277  new_d.clear();
278  throw;
279  }
280  // Move in.
281  *this = std::move(new_d);

393  template <typename It, typename Exponent, insert_enabler<It,Exponent> = 0>
394  void insert(It begin, It end, const Exponent &e)
395  {
396  p_type term;
397  // Assign the exponent.
398  term.e = safe_cast<value_type>(e);
399  if (unlikely(term.e <= 0)) {
400  piranha_throw(std::invalid_argument,"a term of a divisor must have a positive exponent");
401  }
402  // Build the vector to be inserted.
403  for (; begin != end; ++begin) {
404  term.v.push_back(safe_cast<value_type>(*begin));
405  }
406  // Range check.
407  if (unlikely(!term_range_check(term))) {
408  piranha_throw(std::invalid_argument,"an element in a term of a divisor is outside the allowed range");
409  }
410  // Check that the term is canonical.
411  if (unlikely(!term_is_canonical(term))) {
412  piranha_throw(std::invalid_argument,"term not in canonical form");
413  }
414  // Perform the insertion.
415  insertion_impl(std::move(term));
416  }

578  void print(std::ostream &os, const symbol_set &args) const
579  {
580  // Don't print anything if there are no terms.
581  if (m_container.empty()) {
582  return;
583  }
584  if (unlikely(m_container.begin()->v.size() != args.size())) {
585  piranha_throw(std::invalid_argument,"invalid size of arguments set");
586  }
587  const auto it_f = m_container.end();
588  bool first_term = true;
589  os << "1/[";
590  for (auto it = m_container.begin(); it != it_f; ++it) {
591  // If this is not the first term, print a leading '*' operator.
592  if (first_term) {
593  first_term = false;
594  } else {
595  os << '*';
596  }
597  bool printed_something = false;
598  os << '(';
599  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i) {
600  // If the aij is zero, don't print anything.
601  if (math::is_zero(it->v[i])) {
602  continue;
603  }
604  // A positive aij, in case previous output exists, must be preceded
605  // by a "+" sign.
606  if (it->v[i] > 0 && printed_something) {
607  os << '+';
608  }
609  // Print the aij, unless it's "-1": in that case, just print the minus sign.
610  if (it->v[i] == -1) {
611  os << '-';
612  } else if (it->v[i] != 1) {
613  os << detail::prepare_for_print(it->v[i]) << '*';
614  }
615  // Finally, print name of variable.
616  os << args[i].get_name();
617  printed_something = true;
618  }
619  os << ')';
620  // Print the exponent, if different from one.
621  if (it->e != 1) {
622  os << "**" << detail::prepare_for_print(it->e);
623  }
624  }
625  os << ']';
626  }

700  template <typename U>
701  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
702  {
703  // Just return 1 if the container is empty.
704  eval_type<U> retval(1);
705  if (m_container.empty()) {
706  return retval;
707  }
708  // NOTE: the positions map must have the same number of elements as this, and it must
709  // be made of consecutive positions [0,size - 1].
710  if (unlikely(pmap.size() != m_container.begin()->v.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
711  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
712  }
713  // Args must be compatible with this.
714  if (unlikely(args.size() != m_container.begin()->v.size())) {
715  piranha_throw(std::invalid_argument,"invalid size of arguments set");
716  }
717  const auto it_f = m_container.end();
718  for (auto it = m_container.begin(); it != it_f; ++it) {
719  eval_sum_type<U> tmp(0);
720  auto map_it = pmap.begin();
721  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i, ++map_it) {
722  piranha_assert(map_it != pmap.end() && map_it->first == i);
723  // NOTE: might use multadd here eventually for performance.
724  tmp += it->v[i] * map_it->second;
725  }
726  piranha_assert(map_it == pmap.end());
727  // NOTE: consider rewriting this in terms of multiplications as a performance
728  // improvement - the eval_type deduction should be changed accordingly.
729  retval /= math::pow(tmp,it->e);
730  }
731  return retval;
732  }

757  static void multiply(std::array<term<Cf,divisor>,multiply_arity> &res, const term<Cf,divisor> &t1,

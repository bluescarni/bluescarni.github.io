117  // Node class for bucket element.
118  struct node
119  {
120  typedef typename std::aligned_storage<sizeof(T),alignof(T)>::type storage_type;
121  node() : m_next(nullptr) {}
122  // Erase all other ctors/assignments, we do not want to
123  // copy around this object as m_storage is not what it is
124  // (and often could be uninitialized, which would lead to UB if used).
125  node(const node &) = delete;
126  node(node &&) = delete;
127  node &operator=(const node &) = delete;
128  node &operator=(node &&) = delete;
129  // NOTE: the idea here is that we use these helpers only *after* the object has been constructed,
130  // for constructing the object we must always access m_storage directly. The chain of two casts
131  // seems to be the only standard-conforming way of getting out a pointer to T.
132  // http://stackoverflow.com/questions/1082378/what-is-the-basic-use-of-aligned-storage
133  // http://stackoverflow.com/questions/13466556/aligned-storage-and-strict-aliasing
134  // http://en.cppreference.com/w/cpp/types/aligned_storage
135  // A couple of further notes:
136  // - pointer casting to/from void * is ok (4.10.2 and 5.2.9.13, via static_cast);
137  // - the placement new operator (which we use to construct the object into the storage)
138  // is guaranteed to construct the object at the beginning of the storage, and taking
139  // a void * out of the storage will return the address of the object stored within;
140  // - the lifetime of the original POD storage_type ends when we reuse its storage via
141  // placement new.
142  // This page contains some more info about storage reuse:
143  // http://en.cppreference.com/w/cpp/language/lifetime#Storage_reuse
144  const T *ptr() const
145  {
146  piranha_assert(m_next);
147  return static_cast<const T *>(static_cast<const void *>(&m_storage));
148  }
149  T *ptr()
150  {
151  piranha_assert(m_next);
152  return static_cast<T *>(static_cast<void *>(&m_storage));
153  }
154  storage_type m_storage;
155  node *m_next;
156  };

210  {
211  try {
212  auto cur = &m_node;
213  auto other_cur = &other.m_node;
214  while (other_cur->m_next) {
215  if (cur->m_next) {
216  // This assert means we are operating on the last element
217  // of the list, as we are doing back-insertions.
218  piranha_assert(cur->m_next == &terminator);
219  // Create a new node with content equal to other_cur
220  // and linking forward to the terminator.
221  std::unique_ptr<node> new_node(::new node());
222  ::new ((void *)&new_node->m_storage) T(*other_cur->ptr());
223  new_node->m_next = &terminator;
224  // Link the new node.
225  cur->m_next = new_node.release();

227  } else {
228  // This means this is the first node.
229  ::new ((void *)&cur->m_storage) T(*other_cur->ptr());

233  }

260  void steal_from_rvalue(list &&other)
261  {
262  piranha_assert(empty());
263  // Do something only if there is content in the other.
264  if (other.m_node.m_next) {
265  // Move construct current first node with first node of other.
266  ::new ((void *)&m_node.m_storage) T(std::move(*other.m_node.ptr()));
267  // Link remaining content of other into this.
268  m_node.m_next = other.m_node.m_next;
269  // Destroy first node of other.
270  other.m_node.ptr()->~T();
271  other.m_node.m_next = nullptr;
272  }
273  piranha_assert(other.empty());

313  void destroy()
314  {
315  node *cur = &m_node;
316  while (cur->m_next) {
317  // Store the current value temporarily.
318  auto old = cur;
319  // Assign the next.
320  cur = cur->m_next;
321  // Destroy the old payload and erase connections.
322  old->ptr()->~T();
323  old->m_next = nullptr;
324  // If the old node was not the initial one, delete it.
325  if (old != &m_node) {
326  ::delete old;
327  }
328  }
329  // After destruction, the list should be equivalent to a default-constructed one.
330  piranha_assert(empty());
331  }

407  void init_from_n_buckets(const size_type &n_buckets, unsigned n_threads)
408  {
409  piranha_assert(!m_container && !m_log2_size && !m_n_elements);
410  if (unlikely(!n_threads)) {
411  piranha_throw(std::invalid_argument,"the number of threads must be strictly positive");
412  }
413  // Proceed to actual construction only if the requested number of buckets is nonzero.
414  if (!n_buckets) {
415  return;
416  }
417  const size_type log2_size = get_log2_from_hint(n_buckets);
418  const size_type size = size_type(1u) << log2_size;
419  auto new_ptr = m_allocator.allocate(size);
420  if (unlikely(!new_ptr)) {
421  piranha_throw(std::bad_alloc,);
422  }
423  if (n_threads == 1u) {
424  // Default-construct the elements of the array.
425  // NOTE: this is a noexcept operation, no need to account for rolling back.
426  for (size_type i = 0u; i < size; ++i) {
427  m_allocator.construct(&new_ptr[i]);
428  }
429  } else {
430  // Sync variables.
431  using crs_type = std::vector<std::pair<size_type,size_type>>;
432  crs_type constructed_ranges(static_cast<typename crs_type::size_type>(n_threads),std::make_pair(0u,0u));
433  if (unlikely(constructed_ranges.size() != n_threads)) {
434  piranha_throw(std::bad_alloc,);
435  }
436  // Thread function.
437  auto thread_function = [this,new_ptr,&constructed_ranges](const size_type &start, const size_type &end, const unsigned &thread_idx) {
438  for (size_type i = start; i != end; ++i) {
439  this->m_allocator.construct(&new_ptr[i]);
440  }
441  constructed_ranges[thread_idx] = std::make_pair(start,end);
442  };
443  // Work per thread.
444  const auto wpt = size / n_threads;
445  future_list<decltype(thread_pool::enqueue(0u,thread_function,0u,0u,0u))> f_list;
446  try {
447  for (unsigned i = 0u; i < n_threads; ++i) {
448  const auto start = static_cast<size_type>(wpt * i),
449  end = static_cast<size_type>((i == n_threads - 1u) ? size : wpt * (i + 1u));
450  f_list.push_back(thread_pool::enqueue(i,thread_function,start,end,i));
451  }
452  f_list.wait_all();
453  // NOTE: no need to get_all() here, as we know no exceptions will be generated inside thread_func.
454  } catch (...) {
455  // NOTE: everything in thread_func is noexcept, if we are here the exception was thrown by enqueue or push_back.
456  // Wait for everything to wind down.
457  f_list.wait_all();
458  // Destroy what was constructed.
459  for (const auto &r: constructed_ranges) {
460  for (size_type i = r.first; i != r.second; ++i) {
461  m_allocator.destroy(&new_ptr[i]);
462  }
463  }
464  // Deallocate before re-throwing.
465  m_allocator.deallocate(new_ptr,size);
466  throw;
467  }
468  }
469  // Assign the members.
470  m_container = new_ptr;
471  m_log2_size = log2_size;
472  }
473  // Destroy all elements and deallocate m_container.

1002  void rehash(const size_type &new_size, unsigned n_threads = 1u)
1003  {
1004  if (unlikely(!n_threads)) {
1005  piranha_throw(std::invalid_argument,"the number of threads must be strictly positive");
1006  }
1007  // If rehash is requested to zero, do something only if there are no items stored in the table.
1008  if (!new_size) {
1009  if (!size()) {
1010  clear();
1011  }
1012  return;
1013  }
1014  // Do nothing if rehashing to the new size would lead to exceeding the max load factor.
1015  if (static_cast<double>(size()) / static_cast<double>(new_size) > max_load_factor()) {
1016  return;
1017  }
1018  // Create a new table with needed amount of buckets.
1019  hash_set new_table(new_size,m_hasher,m_key_equal,n_threads);
1020  try {
1021  const auto it_f = _m_end();
1022  for (auto it = _m_begin(); it != it_f; ++it) {
1023  const auto new_idx = new_table._bucket(*it);
1024  new_table._unique_insert(std::move(*it),new_idx);
1025  }
1026  } catch (...) {
1027  // Clear up both this and the new table upon any kind of error.
1028  clear();
1029  new_table.clear();
1030  throw;
1031  }
1032  // Retain the number of elements.
1033  new_table.m_n_elements = m_n_elements;
1034  // Clear the old table.
1035  clear();
1036  // Assign the new table.
1037  *this = std::move(new_table);
1038  }

1250  local_iterator _erase(const iterator &it)
1251  {
1252  // Verify the iterator is valid.
1253  piranha_assert(it.m_set == this);
1254  piranha_assert(it.m_idx < bucket_count());
1255  piranha_assert(!m_container[it.m_idx].empty());
1256  piranha_assert(it.m_it != m_container[it.m_idx].end());
1257  auto &bucket = m_container[it.m_idx];
1258  // If the pointed-to element is the first one in the bucket, we need special care.
1259  if (&*it == &*bucket.m_node.ptr()) {
1260  // Destroy the payload.
1261  bucket.m_node.ptr()->~T();
1262  if (bucket.m_node.m_next == &bucket.terminator) {
1263  // Special handling if this was the only element.
1264  bucket.m_node.m_next = nullptr;
1265  return bucket.end();
1266  } else {
1267  // Store the link in the second element.
1268  auto tmp = bucket.m_node.m_next->m_next;
1269  // Move-construct from the second element, and then destroy it.
1270  ::new ((void *)&bucket.m_node.m_storage) T(std::move(*bucket.m_node.m_next->ptr()));
1271  bucket.m_node.m_next->ptr()->~T();
1272  ::delete bucket.m_node.m_next;
1273  // Establish the new link.
1274  bucket.m_node.m_next = tmp;
1275  return bucket.begin();
1276  }
1277  } else {
1278  auto b_it = bucket.begin();
1279  auto prev_b_it = b_it;
1280  const auto b_it_f = bucket.end();
1281  ++b_it;
1282  for (; b_it != b_it_f; ++b_it, ++prev_b_it) {
1283  if (&*b_it == &*it) {
1284  // Assign to the previous element the next link of the current one.
1285  prev_b_it.m_ptr->m_next = b_it.m_ptr->m_next;
1286  // Delete the current one.
1287  b_it.m_ptr->ptr()->~T();
1288  ::delete b_it.m_ptr;
1289  break;
1290  };
1291  }
1292  // We never want to go through the whole list, it means the element
1293  // to which 'it' refers is not here: assert that the iterator we just
1294  // erased was not end() - i.e., it was pointing to something.
1295  piranha_assert(b_it.m_ptr);
1296  // See if the erased iterator was the last one of the list.
1297  const auto tmp = prev_b_it;
1298  ++prev_b_it;
1299  if (prev_b_it == b_it_f) {
1300  // Iterator is the last one, return local end().
1301  return b_it_f;
1302  } else {
1303  // Iterator is not the last one, return it.
1304  return tmp;
1305  }
1306  }

1314  if (environment::shutdown()) {
1315  return true;
1316  }

104  using size_type = typename detail::static_vector_size_type<MaxSize>::type;
105  private:
106 #if !defined(PIRANHA_DOXYGEN_INVOKED)
107  PIRANHA_TT_CHECK(is_container_element,T);
108  // This check is against overflows when using memcpy.
109  static_assert(std::numeric_limits<size_type>::max() <= std::numeric_limits<std::size_t>::max() / sizeof(T),
110  "The size type for static_vector might overflow.");
111  // Check for overflow in the definition of the storage type.
112  static_assert(MaxSize < std::numeric_limits<std::size_t>::max() / sizeof(T),"Overflow in the computation of storage size.");
113  // NOTE: some notes about the use of raw storage, after some research in the standard:
114  // - storage type is guaranteed to be a POD able to hold any object whose size is at most Len
115  // and alignment a divisor of Align (20.9.7.6). Thus, we can store on object of type T at the
116  // beginning of the storage;
117  // - POD types are trivially copyable, and thus they occupy contiguous bytes of storage (1.8/5);
118  // - the meaning of "contiguous" seems not to be explicitly stated, but it can be inferred
119  // (e.g., 23.3.2.1/1) that it has to do with the fact that one can do pointer arithmetics
120  // to move around in the storage;
121  // - the footnote in 5.7/6 implies that casting around to different pointer types and then doing
122  // pointer arithmetics has the expected behaviour of moving about with discrete offsets equal to the sizeof
123  // the pointed-to object (i.e., it seems like pointer arithmetics does not suffer from
124  // strict aliasing and type punning issues - at least as long as one goest through void *
125  // conversions to get the address of the storage - note also 3.9.2/3). This supposedly applies to arrays only,
126  // but, again, it is implied in other places that it also holds true for contiguous storage;
127  // - the no-padding guarantee for arrays (which also consist of contiguous storage) implies that sizeof must be
128  // a multiple of the alignment for a given type;
129  // - the definition of alignment (3.11) implies that if one has contiguous storage starting at an address in
130  // which T can be constructed, the offsetting the initial address by multiples of the alignment value will
131  // still produce addresses at which the object can be constructed;
132  // - in general, we are assuming here that we can handle contiguous storage the same way arrays can be handled (e.g.,
133  // to get the end() iterator we get one past the last element);
134  // - note that placement new will work as expected (i.e., it will construct the object exactly at the address passed
135  // in as parameter).
136  typedef typename std::aligned_storage<sizeof(T) * MaxSize,alignof(T)>::type storage_type;
137 #endif

161  static_vector(const static_vector &other):m_tag(1u),m_size(0u)
162  {
163  // NOTE: here and elsewhere, the standard implies (3.9/2) that we can use this optimisation
164  // for trivially copyable types. GCC does not support the type trait yet, so we restrict the
165  // optimisation to POD types (which are trivially copyable).
166  if (std::is_pod<T>::value) {
167  std::memcpy(vs(),other.vs(),other.m_size * sizeof(T));
168  m_size = other.m_size;
169  } else {
170  try {
171  const auto size = other.size();
172  for (size_type i = 0u; i < size; ++i) {
173  push_back(other[i]);
174  }
175  } catch (...) {
176  // Roll back the constructed items before re-throwing.
177  destroy_items();
178  throw;
179  }
180  }
181  }

261  static_vector &operator=(static_vector &&other) noexcept
262  {
263  if (likely(this != &other)) {
264  if (std::is_pod<T>::value) {
265  std::memcpy(vs(),other.vs(),other.m_size * sizeof(T));
266  } else {
267  const size_type old_size = m_size, new_size = other.m_size;
268  if (old_size <= new_size) {
269  // Move assign new content into old content.
270  for (size_type i = 0u; i < old_size; ++i) {
271  (*this)[i] = std::move(other[i]);
272  }
273  // Move construct remaining elements.
274  for (size_type i = old_size; i < new_size; ++i) {
275  ::new (static_cast<void *>(ptr() + i)) value_type(std::move(other[i]));
276  }
277  } else {
278  // Move assign new content into old content.
279  for (size_type i = 0u; i < new_size; ++i) {
280  (*this)[i] = std::move(other[i]);
281  }
282  // Destroy excess content.
283  for (size_type i = new_size; i < old_size; ++i) {
284  ptr()[i].~T();
285  }
286  }
287  }
288  m_size = other.m_size;
289  // Nuke the other.
290  other.destroy_items();
291  other.m_size = 0u;
292  }
293  return *this;
294  }

448  void resize(const size_type &new_size)
449  {
450  if (unlikely(new_size > MaxSize)) {
451  piranha_throw(std::bad_alloc,);
452  }
453  const auto old_size = m_size;
454  if (new_size == old_size) {
455  return;
456  } else if (new_size > old_size) {
457  size_type i = old_size;
458  // Construct new in case of larger size.
459  try {
460  for (; i < new_size; ++i) {
461  // NOTE: placement-new syntax for value initialization, the same as performed by
462  // std::vector (see 23.3.6.2 and 23.3.6.3/9).
463  // http://en.cppreference.com/w/cpp/language/value_initialization
464  ::new (static_cast<void *>(ptr() + i)) value_type();
465  piranha_assert(m_size != MaxSize);
466  ++m_size;
467  }
468  } catch (...) {
469  for (size_type j = old_size; j < i; ++j) {
470  ptr()[j].~T();
471  piranha_assert(m_size);
472  --m_size;
473  }
474  throw;
475  }
476  } else {
477  // Destroy in case of smaller size.
478  for (size_type i = new_size; i < old_size; ++i) {
479  ptr()[i].~T();
480  piranha_assert(m_size);
481  --m_size;
482  }
483  }

331  bool is_compatible(const symbol_set &args) const noexcept
332  {
333  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
334  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
335  // in other methods.
336  const auto s = args.size();
337  // No args means the value must also be zero.
338  if (s == 0u) {
339  return !m_value;
340  }
341  const auto &limits = ka::get_limits();
342  // If we overflow the maximum size available, we cannot use this object as key in series.
343  if (s >= limits.size()) {
344  return false;
345  }
346  const auto &l = limits[s];
347  // Value is compatible if it is within the bounds for the given size.
348  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
349  }

477  static void multiply(std::array<term<Cf,kronecker_monomial>,multiply_arity> &res, const term<Cf,kronecker_monomial> &t1,
478  const term<Cf,kronecker_monomial> &t2, const symbol_set &args)
479  {
480  // NOTE: this method is actually never called as the specialised Kronecker polynomial
481  // multiplier does not use it. It is here for completeness and because the series multiplier
482  // for k polynomial currently inherits from the base multiplier, which needs this.
483  // Coefficient first.
484  res[0u].m_cf = t1.m_cf * t2.m_cf;
485  // Now the key.

490  result.push_back(tmp1[i]);

537  {
538  const auto v = unpack(args);
539  const auto size = args.size();
540  decltype(args.size()) n_linear = 0u, candidate = 0u;
541  for (typename v_type::size_type i = 0u; i < size; ++i) {
542  integer tmp;

648  {
649  const auto tmp = unpack(args);
650  std::ostringstream oss_num, oss_den, *cur_oss;
651  const value_type zero(0), one(1);

692  std::pair<T,kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
693  {
694  auto v = unpack(args);
695  // Cannot take derivative wrt more than one variable, and the position of that variable
696  // must be compatible with the monomial.
697  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
698  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
699  }
700  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
701  // variable with zero exponent.
702  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
703  // there is a valid position and hence the size must be not zero.
704  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
705  return std::make_pair(T(0),kronecker_monomial(args));
706  }
707  auto v_b = v.begin();
708  // Original exponent.
709  T n(v_b[*p.begin()]);
710  // Decrement the exponent in the monomial.
711  if (unlikely(n == std::numeric_limits<T>::min())) {
712  piranha_throw(std::invalid_argument,"negative overflow error in the calculation of the "
713  "partial derivative of a monomial");
714  }
715  v_b[*p.begin()] = static_cast<T>(n - T(1));
716  kronecker_monomial tmp_km;
717  tmp_km.m_value = ka::encode(v);
718  return std::make_pair(n,std::move(tmp_km));
719  }

742  std::pair<integer,kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
743  {
744  v_type v = unpack(args), retval;
745  value_type expo(0), one(1);
746  for (min_int<typename v_type::size_type,decltype(args.size())> i = 0u; i < args.size(); ++i) {
747  if (math::is_zero(expo) && s < args[i]) {
748  // If we went past the position of s in args and still we
749  // have not performed the integration, it means that we need to add
750  // a new exponent.
751  retval.push_back(one);
752  expo = one;
753  }
754  retval.push_back(v[i]);
755  if (args[i] == s) {
756  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
757  // we are never going to land here as args[i] is at this point never going to be s.
758  retval[i] = static_cast<value_type>(integer(retval[i]) + 1);
759  if (math::is_zero(retval[i])) {
760  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
761  }
762  expo = retval[i];
763  }
764  }
765  // If expo is still zero, it means we need to add a new exponent at the end.
766  if (math::is_zero(expo)) {
767  retval.push_back(one);
768  expo = one;
769  }
770  return std::make_pair(integer(expo),kronecker_monomial(ka::encode(retval)));
771  }

797  template <typename U>
798  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
799  {
800  using return_type = eval_type<U>;
801  using size_type = typename v_type::size_type;
802  // NOTE: here we can check the pmap size only against args.
803  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
804  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
805  }
806  auto v = unpack(args);
807  return_type retval(1);
808  auto it = pmap.begin();
809  for (min_int<size_type,decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
810  piranha_assert(it != pmap.end() && it->first == i);
811  retval *= math::pow(it->second,v[i]);
812  }
813  piranha_assert(it == pmap.end());

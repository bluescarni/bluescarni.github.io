350  void trim_identify(symbol_set &candidates, const symbol_set &args) const
351  {
352  if (unlikely(m_container.size() != args.size())) {
353  piranha_throw(std::invalid_argument,"invalid arguments set for trim_identify()");
354  }
355  for (size_type i = 0u; i < m_container.size(); ++i) {
356  if (!math::is_zero(m_container[i]) && std::binary_search(candidates.begin(),candidates.end(),
357  args[static_cast<symbol_set::size_type>(i)]))
358  {
359  candidates.remove(args[static_cast<symbol_set::size_type>(i)]);
360  }
361  }

376  Derived trim(const symbol_set &trim_args, const symbol_set &orig_args) const
377  {
378  if (unlikely(m_container.size() != orig_args.size())) {
379  piranha_throw(std::invalid_argument,"invalid arguments set for trim()");
380  }
381  Derived retval;
382  for (size_type i = 0u; i < m_container.size(); ++i) {
383  if (!std::binary_search(trim_args.begin(),trim_args.end(),
384  orig_args[static_cast<symbol_set::size_type>(i)]))
385  {
386  retval.push_back(m_container[i]);
387  }
388  }

431  bool is_compatible(const symbol_set &args) const noexcept
432  {
433  const auto s = args.size();
434  // No args means the value must also be zero.
435  if (s == 0u) {
436  return !m_value;
437  }
438  const auto &limits = ka::get_limits();
439  // If we overflow the maximum size available, we cannot use this object as key in series.
440  if (s >= limits.size()) {
441  return false;
442  }
443  const auto &l = limits[s];
444  // Value is compatible if it is within the bounds for the given size.
445  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
446  return false;
447  }
448  // Now check for the first multiplier.
449  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
450  // we do not need to put @throw specifications in the doc.
451  const auto unpacked = unpack(args);
452  // We know that s != 0.
453  piranha_assert(unpacked.size() > 0u);
454  for (typename v_type::size_type i = 0u; i < s; ++i) {
455  if (unpacked[i] < value_type(0)) {
456  return false;
457  } else if (unpacked[i] > value_type(0)) {
458  break;
459  }
460  }
461  return true;
462  }

664  template <typename Cf, multiply_enabler<Cf> = 0>
665  static void multiply(std::array<term<Cf,real_trigonometric_kronecker_monomial>,multiply_arity> &res,
666  const term<Cf,real_trigonometric_kronecker_monomial> &t1, const term<Cf,real_trigonometric_kronecker_monomial> &t2,
667  const symbol_set &args)
668  {
669  // Coefficients first.
670  // NOTE: first divide by two here and then multiply, at least for series? Would reduce
671  // quadratically the number of coefficient divisions.
672  // Similarly, it might be worth to change the sign as needed before the multiplication.
673  // NOTE: and if we divide everything by two in the end instead? Might be better overall.
674  Cf res_cf(t1.m_cf * t2.m_cf);
675  res_cf /= 2;
676  res[0u].m_cf = res_cf;
677  res[1u].m_cf = std::move(res_cf);
678  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
679  if (f1 && f2) {
680  // cos, cos: no change.
681  } else if (!f1 && !f2) {
682  // sin, sin: negate the plus.
683  math::negate(res[0u].m_cf);
684  } else if (!f1 && f2) {
685  // sin, cos: no change.
686  } else {
687  // cos, sin: negate the minus.
688  math::negate(res[1u].m_cf);
689  }
690  // Now the keys.
691  auto &retval_plus = res[0u].m_key;
692  auto &retval_minus = res[1u].m_key;
693  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
694  bool sign_plus = false, sign_minus = false;
695  const auto size = args.size();
696  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
697  v_type result_plus, result_minus;
698  for (typename v_type::size_type i = 0u; i < size; ++i) {
699  result_plus.push_back(tmp1[i]);
700  detail::km_safe_adder(result_plus[i],tmp2[i]);
701  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
702  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
703  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
704  // to int and safe_adder won't work as it expects identical types.
705  result_minus.push_back(tmp1[i]);
706  detail::km_safe_adder(result_minus[i],static_cast<value_type>(-tmp2[i]));
707  }
708  // Handle sign changes.
709  sign_plus = canonicalise_impl(result_plus);
710  sign_minus = canonicalise_impl(result_minus);
711  // Compute them before assigning, so in case of exceptions we do not touch the return values.
712  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
713  retval_plus.m_value = re_plus;
714  retval_minus.m_value = re_minus;
715  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
716  retval_plus.m_flavour = f;
717  retval_minus.m_flavour = f;
718  // If multiplier sign was changed and the result is a sine, negate the coefficient.
719  if (sign_plus && !res[0u].m_key.get_flavour()) {
720  math::negate(res[0u].m_cf);
721  }
722  if (sign_minus && !res[1u].m_key.get_flavour()) {
723  math::negate(res[1u].m_cf);
724  }
725  }

780  void print(std::ostream &os, const symbol_set &args) const
781  {
782  // Don't print anything in case all multipliers are zero.
783  if (m_value == value_type(0)) {
784  return;
785  }
786  if (m_flavour) {
787  os << "cos(";
788  } else {
789  os << "sin(";
790  }
791  const auto tmp = unpack(args);
792  piranha_assert(tmp.size() == args.size());
793  const value_type zero(0), one(1), m_one(-1);
794  bool empty_output = true;
795  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
796  if (tmp[i] != zero) {
797  // A positive multiplier, in case previous output exists, must be preceded
798  // by a "+" sign.
799  if (tmp[i] > zero && !empty_output) {
800  os << '+';
801  }
802  // Print the multiplier, unless it's "-1": in that case, just print the minus sign.
803  if (tmp[i] == m_one) {
804  os << '-';
805  } else if (tmp[i] != one) {
806  os << detail::prepare_for_print(tmp[i]) << '*';
807  }
808  // Finally, print name of variable.
809  os << args[i].get_name();
810  empty_output = false;
811  }
812  }
813  os << ")";
814  }

878  std::pair<T,real_trigonometric_kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const

880  auto v = unpack(args);
881  // Cannot take derivative wrt more than one variable, and the position of that variable
882  // must be compatible with the monomial.
883  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
884  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
885  }
886  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
887  // variable with zero multiplier.
888  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
889  // there is a valid position and hence the size must be not zero.
890  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
891  return std::make_pair(T(0),real_trigonometric_kronecker_monomial(args));
892  }
893  auto v_b = v.begin();
894  // Original multiplier.
895  T n(v_b[*p.begin()]);
896  // Create a copy of this.
897  real_trigonometric_kronecker_monomial tmp_m(*this);
898  // Flip the flavour.
899  tmp_m.set_flavour(!get_flavour());
900  // Flip the sign of the multiplier as needed.
901  if (get_flavour()) {
902  // NOTE: this is safe, as it is coming out of a k codification which
903  // is symmetric.
904  math::negate(n);
905  }
906  return std::make_pair(std::move(n),std::move(tmp_m));
907  }

927  std::pair<integer,real_trigonometric_kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
928  {

930  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i) {
931  if (args[i] == s && !math::is_zero(v[i])) {

937  if (!get_flavour()) {

939  }
940  return std::make_pair(std::move(tmp_n),std::move(tmp_m));

943  }

970  template <typename U>
971  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
972  {
973  using return_type = eval_type<U>;
974  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
975  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
976  }
977  // Run the unpack before the checks below in order to check the suitability of args.
978  auto v = unpack(args);
979  if (args.size() == 0u) {
980  if (get_flavour()) {
981  return return_type(1);
982  } else {
983  return return_type(0);
984  }
985  }
986  using tmp_type = decltype(std::declval<U const &>() * std::declval<value_type const &>());
987  tmp_type tmp(0);
988  auto it = pmap.begin();
989  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i, ++it) {
990  piranha_assert(it != pmap.end() && it->first == i);
991  // NOTE: here it might make sense to use multiply_accumulate. There might be a perf gain
992  // for things like real. Maybe fast FMA on Haswell too. Remember to adapt the type requirements
993  // in case we implement this.
994  tmp += it->second * v[i];
995  }
996  piranha_assert(it == pmap.end());
997  if (get_flavour()) {
998  return math::cos(tmp);
999  }
1000  return math::sin(tmp);
1001  }

1046  std::vector<std::pair<subs_type<U>,real_trigonometric_kronecker_monomial>>
1047  subs(const std::string &s, const U &x, const symbol_set &args) const

1109  template <typename U>
1110  std::vector<std::pair<t_subs_type<U>,
1111  real_trigonometric_kronecker_monomial>> t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1112  {
1113  typedef decltype(this->t_subs(name,c,s,args)) ret_type;
1114  typedef typename ret_type::value_type::first_type res_type;
1115  const auto v = unpack(args);
1116  v_type new_v;
1117  value_type n(0);
1118  // Build the new vector key.
1119  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1120  if (args[i].get_name() == name) {
1121  new_v.push_back(value_type(0));
1122  n = v[i];
1123  } else {
1124  new_v.push_back(v[i]);
1125  }
1126  }
1127  // Absolute value of the multiplier.
1128  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1129  // Prepare the powers of c and s to be used in the multiple angles formulae.
1130  std::unordered_map<value_type,U> c_map, s_map;
1131  c_map[0] = U(1);
1132  s_map[0] = U(1);
1133  for (value_type k(0); k < abs_n; ++k) {
1134  c_map[k + value_type(1)] = c_map[k] * c;
1135  s_map[k + value_type(1)] = s_map[k] * s;
1136  }
1137  // Init with the first element in the summation.
1138  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n])),
1139  sin_nx((sin_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1140  for (value_type k(0); k < abs_n; ++k) {
1141  const value_type p = abs_n - (k + value_type(1));
1142  piranha_assert(p >= value_type(0));
1143  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as long
1144  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1145  // left-to-right associativity of multiplication.
1146  res_type tmp(math::binomial(abs_n,k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1147  cos_nx += cos_phase(p) * tmp;
1148  sin_nx += sin_phase(p) * tmp;
1149  }
1150  // Change sign as necessary.
1151  if (abs_n != n) {
1152  math::negate(sin_nx);
1153  }
1154  // Buld the new keys and canonicalise as needed.
1155  const bool sign_changed = canonicalise_impl(new_v);
1156  piranha_assert(new_v.size() == v.size());
1157  const auto new_int = ka::encode(new_v);
1158  real_trigonometric_kronecker_monomial cos_key(new_int,true), sin_key(new_int,false);
1159  ret_type retval;
1160  if (get_flavour()) {
1161  retval.push_back(std::make_pair(std::move(cos_nx),std::move(cos_key)));
1162  retval.push_back(std::make_pair(std::move(sin_nx),std::move(sin_key)));
1163  // Need to flip the sign on the sin * sin product if sign was not changed.
1164  if (!sign_changed) {
1165  math::negate(retval[1u].first);
1166  }
1167  } else {
1168  retval.push_back(std::make_pair(std::move(sin_nx),std::move(cos_key)));
1169  retval.push_back(std::make_pair(std::move(cos_nx),std::move(sin_key)));
1170  // Need to flip the sign on the cos * sin product if sign was changed.
1171  if (sign_changed) {
1172  math::negate(retval[1u].first);
1173  }
1174  }
1175  return retval;
1176  }

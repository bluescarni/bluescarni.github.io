79  template <typename Term, typename T>
80  struct subs_term_score
81  {
82  static const unsigned value = static_cast<unsigned>(has_ipow_subs<typename Term::cf_type,T>::value) +
83  (static_cast<unsigned>(key_has_ipow_subs<typename Term::key_type,T>::value) << 1u);
84  };
85  // Case 1: subs only on cf.
86  template <typename T, typename Term>
87  using cf_subs_type = decltype(math::ipow_subs(std::declval<typename Term::cf_type const &>(),std::declval<std::string const &>(),
88  std::declval<const integer &>(),std::declval<T const &>()));
89  template <typename T, typename Term>
90  using ret_type_1 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<Derived const &>());
91  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 1u,int>::type = 0>
92  static ret_type_1<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
93  {
94  Derived tmp;
95  tmp.set_symbol_set(s_set);
96  tmp.insert(Term(typename Term::cf_type(1),t.m_key));
97  // NOTE: use moves here in case the multiplication can take advantage.
98  return math::ipow_subs(t.m_cf,name,n,x) * std::move(tmp);
99  }
100  // Case 2: subs only on key.
101  template <typename T, typename Term>
102  using k_subs_type = typename decltype(std::declval<const typename Term::key_type &>().ipow_subs(std::declval<const std::string &>(),
103  std::declval<const integer &>(),std::declval<const T &>(),std::declval<const symbol_set &>()))::value_type::first_type;
104  template <typename T, typename Term>
105  using ret_type_2_ = decltype(std::declval<Derived const &>() * std::declval<const k_subs_type<T,Term> &>());
106  template <typename T, typename Term>
107  using ret_type_2 = typename std::enable_if<is_addable_in_place<ret_type_2_<T,Term>>::value &&
108  std::is_constructible<ret_type_2_<T,Term>,int>::value,ret_type_2_<T,Term>>::type;
109  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 2u,int>::type = 0>
110  static ret_type_2<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
111  {
112  ret_type_2<T,Term> retval(0);
113  auto ksubs = t.m_key.ipow_subs(name,n,x,s_set);
114  for (auto &p: ksubs) {
115  Derived tmp;
116  tmp.set_symbol_set(s_set);
117  tmp.insert(Term{t.m_cf,std::move(p.second)});
118  // NOTE: possible use of multadd here in the future.
119  retval += std::move(tmp) * std::move(p.first);
120  }
121  return retval;
122  }
123  // Case 3: subs on cf and key.
124  // NOTE: the checks on type 2 are already present in the alias above.
125  template <typename T, typename Term>
126  using ret_type_3 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<const ret_type_2<T,Term> &>());
127  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 3u,int>::type = 0>
128  static ret_type_3<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
129  {
130  // Accumulator for the sum below. This is the same type resulting from case 2.
131  ret_type_2<T,Term> acc(0);
132  auto ksubs = t.m_key.ipow_subs(name,n,x,s_set);
133  auto cf_subs = math::ipow_subs(t.m_cf,name,n,x);
134  for (auto &p: ksubs) {
135  Derived tmp;
136  tmp.set_symbol_set(s_set);
137  tmp.insert(Term(typename Term::cf_type(1),std::move(p.second)));
138  // NOTE: multadd chance.
139  acc += std::move(tmp) * std::move(p.first);
140  }
141  return std::move(cf_subs) * std::move(acc);
142  }
143  // Initial definition of the subs type.
144  template <typename T>
145  using subs_type_ = decltype(subs_term_impl(std::declval<typename Series::term_type const &>(),std::declval<std::string const &>(),
146  std::declval<const integer &>(),std::declval<const T &>(),std::declval<symbol_set const &>()));
147  // Enable conditionally based on the common requirements in the ipow_subs() method.
148  template <typename T>
149  using ipow_subs_type = typename std::enable_if<std::is_constructible<subs_type_<T>,int>::value && is_addable_in_place<subs_type_<T>>::value,

151  // Enabler for the alternate overload.

144  template <typename Float>
145  void construct_from_interoperable(const Float &x, typename std::enable_if<std::is_floating_point<Float>::value>::type * = nullptr)
146  {
147  if (unlikely(!std::isfinite(x))) {
148  piranha_throw(std::invalid_argument,"cannot construct a rational from a non-finite floating-point number");
149  }
150  // Denominator is always inited as 1.
151  m_den = 1;
152  if (x == Float(0)) {
153  // m_den is 1 already.
154  return;
155  }
156  Float abs_x = std::abs(x);
157  const unsigned radix = static_cast<unsigned>(std::numeric_limits<Float>::radix);
158  int_type i_part;
159  int exp = std::ilogb(abs_x);
160  while (exp >= 0) {
161  i_part += int_type(radix).pow(exp);
162  const Float tmp = std::scalbn(Float(1),exp);
163  if (unlikely(tmp == HUGE_VAL)) {
164  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
165  }
166  abs_x -= tmp;
167  // Break out if x is an exact integer.
168  if (abs_x == Float(0)) {
169  // m_den is 1 already.
170  m_num = i_part;
171  if (x < Float(0)) {
172  m_num.negate();
173  }
174  return;
175  }
176  exp = std::ilogb(abs_x);
177  if (unlikely(exp == INT_MAX || exp == FP_ILOGBNAN)) {
178  piranha_throw(std::invalid_argument,"error calling std::ilogb");
179  }
180  }
181  piranha_assert(abs_x < Float(1));
182  // Lift up the decimal part into an integer.
183  while (abs_x != Float(0)) {
184  abs_x = std::scalbln(abs_x,1);
185  if (unlikely(abs_x == HUGE_VAL)) {
186  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
187  }
188  const auto t_abs_x = std::trunc(abs_x);
189  m_den *= radix;
190  m_num *= radix;
191  // NOTE: here t_abs_x is guaranteed to be in
192  // [0,radix - 1], so the cast to unsigned should be ok.
193  // Note that floating-point numbers are guaranteed to be able
194  // to represent exactly at least a [-exp,+exp] exponent range
195  // (see the minimum values for the FLT constants in the C standard).
196  m_num += static_cast<unsigned>(t_abs_x);
197  abs_x -= t_abs_x;
198  }
199  math::multiply_accumulate(m_num,i_part,m_den);
200  canonicalise();
201  if (x < Float(0)) {
202  m_num.negate();
203  }
204  }

228  // In-place add.
229  mp_rational &in_place_add(const mp_rational &other)
230  {
231  // NOTE: all this should never throw because we only operate on mp_integer objects,
232  // no conversions involved, etc.
233  if (m_den.is_unitary() && other.m_den.is_unitary()) {
234  // Both are integers, we can just add without canonicalising.
235  // NOTE: this is safe is this and other are the same thing.
236  m_num += other.m_num;
237  } else if (m_den == other.m_den) {
238  // Denominators are the same, add numerators and canonicalise.
239  // NOTE: safe if this and other coincide.
240  m_num += other.m_num;
241  canonicalise();
242  } else {
243  // The general case.
244  // NOTE: here dens are different (cannot coincide), and thus
245  // this and other must be separate objects.
246  m_num *= other.m_den;
247  math::multiply_accumulate(m_num,m_den,other.m_num);
248  m_den *= other.m_den;
249  canonicalise();
250  }
251  return *this;
252  }

54 // Task queue class. Inspired by:
55 // https://github.com/progschj/ThreadPool
56 class task_queue
57 {
58  struct runner
59  {
60  runner(task_queue *ptr, unsigned n):m_ptr(ptr),m_n(n) {}
61  void operator()() const
62  {
63  // Don't stop if we cannot bind.
64  try {
65  thread_management::bind_to_proc(m_n);
66  } catch (...) {
67  // NOTE: logging candidate.
68  }
69  try {
70  while (true) {
71  std::unique_lock<std::mutex> lock(m_ptr->m_mutex);
72  while (!m_ptr->m_stop && m_ptr->m_tasks.empty()) {
73  // Need to wait for something to happen only if the task
74  // list is empty and we are not stopping.
75  // NOTE: wait will be noexcept in C++14.
76  m_ptr->m_cond.wait(lock);
77  }
78  if (m_ptr->m_stop && m_ptr->m_tasks.empty()) {
79  // If the stop flag was set, and we do not have more tasks,
80  // just exit.
81  break;
82  }
83  // NOTE: move constructor of std::function could throw, unfortunately.
84  std::function<void()> task(std::move(m_ptr->m_tasks.front()));
85  m_ptr->m_tasks.pop();
86  lock.unlock();
87  task();
88  }
89  } catch (...) {
90  // The errors we could get here are:
91  // - threading primitives,
92  // - move-construction of std::function,
93  // - queue popping (I guess unlikely, as the destructor of std::function
94  // is noexcept).
95  // In any case, not much that can be done to recover from this, better to abort.
96  // NOTE: logging candidate.
97  std::abort();
98  }
99  // Free the MPFR caches.
100  ::mpfr_free_cache();
101  }
102  task_queue *m_ptr;
103  const unsigned m_n;
104  };
105  public:
106  task_queue(unsigned n):m_stop(false)
107  {
108  m_thread.reset(::new std::thread(runner{this,n}));
109  }
110  ~task_queue()
111  {
112  // NOTE: logging candidate (catch any exception,
113  // log it and abort as there is not much we can do).
114  try {
115  stop();
116  } catch (...) {
117  std::abort();
118  }
119  }
120  template <typename F, typename ... Args>
121  auto enqueue(F &&f, Args && ... args) -> std::future<decltype(f(args...))>
122  {
123  using f_ret_type = decltype(f(args...));
124  using p_task_type = std::packaged_task<f_ret_type()>;
125  auto task = std::make_shared<p_task_type>(std::bind(std::forward<F>(f),std::forward<Args>(args)...));
126  std::future<f_ret_type> res = task->get_future();
127  {
128  std::unique_lock<std::mutex> lock(m_mutex);
129  if (m_stop) {
130  // Enqueueing is not allowed if the queue is stopped.
131  piranha_throw(std::runtime_error,"cannot enqueue task while task queue is stopping");
132  }
133  m_tasks.push([task](){(*task)();});
134  }
135  // NOTE: notify_one is noexcept.
136  m_cond.notify_one();
137  return res;
138  }
139  // NOTE: we call this only from dtor, it is here in order to be able to test it.
140  // So the exception handling in dtor will suffice, keep it in mind if things change.
141  void stop()
142  {
143  {
144  std::unique_lock<std::mutex> lock(m_mutex);
145  if (m_stop) {
146  // Already stopped.
147  return;
148  }
149  m_stop = true;
150  }
151  // Notify the thread that queue has been stopped, wait for it
152  // to consume the remaining tasks and exit.
153  m_cond.notify_one();
154  m_thread->join();
155  }
156  private:
157  bool m_stop;
158  std::condition_variable m_cond;
159  std::mutex m_mutex;
160  std::queue<std::function<void()>> m_tasks;
161  std::unique_ptr<std::thread> m_thread;
162 };

166  std::vector<std::unique_ptr<task_queue>> retval;

268  static void resize(unsigned new_size)
269  {
270  if (unlikely(new_size == 0u)) {
271  piranha_throw(std::invalid_argument,"cannot resize the thread pool to zero");
272  }
273  std::lock_guard<std::mutex> lock(s_mutex);
274  using size_type = decltype(base::s_queues.size());
275  const size_type new_s = static_cast<size_type>(new_size);
276  if (unlikely(new_s != new_size)) {
277  piranha_throw(std::invalid_argument,"invalid size");
278  }
279  std::vector<std::unique_ptr<detail::task_queue>> new_queues;
280  new_queues.reserve(new_s);
281  for (size_type i = 0u; i < new_s; ++i) {
282  new_queues.emplace_back(::new detail::task_queue(static_cast<unsigned>(i)));
283  }
284  // NOTE: here the allocator is not swapped, as std::allocator won't propagate on swap.
285  // Besides, all instances of std::allocator are equal, so the operation is well-defined.
286  // http://en.cppreference.com/w/cpp/container/vector/swap
287  // If an exception gets actually thrown, no big deal.
288  new_queues.swap(base::s_queues);
289  }

310  template <typename Int, typename = typename std::enable_if<(std::is_integral<Int>::value && std::is_unsigned<Int>::value) ||
311  std::is_same<Int,integer>::value>::type>
312  static unsigned use_threads(const Int &work_size, const Int &min_work_per_thread)
313  {
314  // Check input param.
315  if (unlikely(math::is_zero(min_work_per_thread))) {
316  piranha_throw(std::invalid_argument,"invalid value for minimum work per thread");
317  }
318  // Don't use threads if we are not in the main thread.
319  if (std::this_thread::get_id() != runtime_info::get_main_thread_id()) {
320  return 1u;
321  }
322  std::lock_guard<std::mutex> lock(s_mutex);
323  const auto n_threads = static_cast<unsigned>(base::s_queues.size());
324  piranha_assert(n_threads);
325  if (work_size / n_threads >= min_work_per_thread) {
326  // Enough work per thread, use them all.
327  return n_threads;
328  }
329  // Return a number of threads such that each thread consumes at least min_work_per_thread.
330  // Never return 0.
331  return static_cast<unsigned>(std::max(Int(1),static_cast<Int>(work_size / min_work_per_thread)));
332  }

395  {

399  } catch (...) {

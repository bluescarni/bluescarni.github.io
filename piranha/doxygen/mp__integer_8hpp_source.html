78 // Considering an array of UIn (unsigned integrals) of a certain size as a continuous sequence of bits,
79 // read the index-th URet (unsigned integral) that can be extracted from the sequence of bits.
80 // The parameter IBits is the number of upper bits of UIn that should be discarded in the computation
81 // (i.e., not considered as part of the continuous sequence of bits). RBits has the same meaning,
82 // but for the output value: it's the number of upper bits that are not considered as part of the
83 // return type.
84 template <typename URet, unsigned IBits = 0u, unsigned RBits = 0u, typename UIn,
85  typename std::enable_if<!read_uint_opt<URet,IBits,RBits,UIn>::value,int>::type = 0>
86 inline URet read_uint(const UIn *ptr, std::size_t size, std::size_t index)
87 {
88  // We can work only with unsigned integer types.
89  static_assert(std::is_integral<UIn>::value && std::is_unsigned<UIn>::value,"Invalid type.");
90  static_assert(std::is_integral<URet>::value && std::is_unsigned<URet>::value,"Invalid type.");
91  // Bit size of the return type.
92  constexpr unsigned r_bits = static_cast<unsigned>(std::numeric_limits<URet>::digits);
93  // Bit size of the input type.
94  constexpr unsigned i_bits = static_cast<unsigned>(std::numeric_limits<UIn>::digits);
95  // The ignored bits in the input type cannot be larger than or equal to its bit size.
96  static_assert(IBits < i_bits,"Invalid ignored input bits size");
97  // Same for for RBits.
98  static_assert(RBits < r_bits,"Invalid ignored output bits size");
99  // Bits effectively considered in the input type.
100  constexpr unsigned ei_bits = i_bits - IBits;
101  // Bits effectively considered in the output type.
102  constexpr unsigned er_bits = r_bits - RBits;
103  // Index in input array from where we will start reading, and bit index within
104  // that element from which the actual reading will start.
105  // NOTE: we need to protect against multiplication overflows in the upper level routine.
106  std::size_t s_index = static_cast<std::size_t>((er_bits * index) / ei_bits),
107  r_index = static_cast<std::size_t>((er_bits * index) % ei_bits);
108  // Check that we are not going to read past the end.
109  piranha_assert(s_index < size);
110  // Check for null.
111  piranha_assert(ptr != nullptr);
112  // Init the return value as 0.
113  URet retval = 0u;
114  // Bits read so far.
115  unsigned read_bits = 0u;
116  while (s_index < size && read_bits < er_bits) {
117  const unsigned unread_bits = er_bits - read_bits,
118  // This can be different from ei_bits only on the first iteration.
119  available_bits = static_cast<unsigned>(ei_bits - r_index),
120  bits_to_read = (available_bits > unread_bits) ? unread_bits : available_bits;
121  // On the first iteration, we might need to drop the first r_index bits.
122  UIn tmp = static_cast<UIn>(ptr[s_index] >> r_index);
123  // Zero out the bits we don't want to read.
124  tmp = clear_top_bits(tmp,i_bits - bits_to_read);
125  // Accumulate into return value.
126  retval = static_cast<URet>(retval + (static_cast<URet>(tmp) << read_bits));
127  // Update loop variables.
128  read_bits += bits_to_read;
129  ++s_index;
130  r_index = 0u;
131  }
132  piranha_assert(s_index == size || read_bits == er_bits);
133  return retval;
134 }

289  template <typename T, typename std::enable_if<std::is_signed<T>::value && std::is_integral<T>::value,int>::type = 0>
290  bool attempt_direct_construction(T n)
291  {
292  try {
293  if (n >= T(0)) {
294  // We can attempt conversion of a non-negative signed directly into limb_t.
295  m_limbs[0u] = boost::numeric_cast<limb_t>(n);
296  } else if (n >= -safe_abs_sint<T>::value) {
297  // Conversion of a negative signed whose value can be safely negated.
298  m_limbs[0u] = boost::numeric_cast<limb_t>(-n);
299  } else {
300  // We cannot convert safely n to a limb_t in this case.
301  return false;
302  }
303  if (n > T(0)) {
304  // Strictly positive value, size is 1.
305  _mp_size = mpz_size_t(1);
306  }
307  if (n < T(0)) {
308  // Strictly negative value, size is -1.
309  _mp_size = mpz_size_t(-1);
310  }
311  // NOTE: if n is zero, size has been inited to zero already.
312  return true;
313  } catch (...) {
314  return false;
315  }
316  }

318  bool attempt_direct_construction(T n)

332  template <typename Integer>
333  void construct_from_integral(Integer n)
334  {
335  // Don't do anything if n is zero or is the direct construction
336  // into the first limb succeeds.
337  if (!n || likely(attempt_direct_construction(n))) {
338  return;
339  }
340  const auto orig_n = n;
341  limb_t bit_idx = 0;
342  while (n != Integer(0)) {
343  if (bit_idx == limb_bits * 2u) {
344  // Clear out before throwing, as this is used in mp_integer as well.
345  _mp_size = 0;
346  m_limbs[0u] = 0u;
347  m_limbs[1u] = 0u;
348  piranha_throw(std::overflow_error,"insufficient bit width");
349  }
350  // NOTE: in C++11 division will round to zero always (for negative numbers as well).
351  // The bit shift operator >> has implementation defined behaviour if n is signed and negative,
352  // so we do not use it.
353  const Integer quot = static_cast<Integer>(n / Integer(2)), rem = static_cast<Integer>(n % Integer(2));
354  if (rem) {
355  set_bit(bit_idx);
356  }
357  n = quot;
358  ++bit_idx;
359  }
360  fix_sign_ctor(orig_n);
361  }

389  void set_bit(const limb_t &idx)
390  {
391  using size_type = typename limbs_type::size_type;
392  piranha_assert(idx < limb_bits * 2u);
393  // Crossing fingers for compiler optimising this out.
394  const auto quot = static_cast<limb_t>(idx / limb_bits), rem = static_cast<limb_t>(idx % limb_bits);
395  m_limbs[static_cast<size_type>(quot)] = static_cast<limb_t>(m_limbs[static_cast<size_type>(quot)] | static_cast<limb_t>(limb_t(1) << rem));
396  // Update the size if needed. The new size must be at least quot + 1, as we set a bit
397  // in the limb with index quot.
398  const auto new_size = static_cast<mpz_size_t>(quot + 1u);
399  if (_mp_size < 0) {
400  if (-new_size < _mp_size) {
401  _mp_size = -new_size;
402  }
403  } else {
404  if (new_size > _mp_size) {
405  _mp_size = new_size;
406  }
407  }

434  template <typename T = static_integer, typename = void>
435  class static_mpz_view
436  {
437  public:
438  // Safe, checked above.
439  static const auto max_tot_nbits = 2u * T::limb_bits;
440  // Check the conversion below.
441  static_assert(max_tot_nbits / unsigned(GMP_NUMB_BITS) + 1u <= std::numeric_limits<std::size_t>::max(),
442  "Overflow error.");
443  // Number of GMP limbs to use.
444  static const std::size_t max_n_gmp_limbs = static_cast<std::size_t>(
445  max_tot_nbits % unsigned(GMP_NUMB_BITS) == 0u ?
446  max_tot_nbits / unsigned(GMP_NUMB_BITS) :
447  max_tot_nbits / unsigned(GMP_NUMB_BITS) + 1u);
448  static_assert(max_n_gmp_limbs >= 1u,"Invalid number of GMP limbs.");
449  static_mpz_view(): m_mpz(),m_limbs() {}
450  explicit static_mpz_view(const static_integer &n)
451  {
452  std::size_t asize;
453  bool sign;
454  if (n._mp_size < 0) {
455  sign = false;
456  asize = static_cast<std::size_t>(-n._mp_size);
457  } else {
458  sign = true;
459  asize = static_cast<std::size_t>(n._mp_size);
460  }
461  piranha_assert(asize <= 2u);
462  const auto tot_nbits = asize * T::limb_bits;
463  const std::size_t n_gmp_limbs = static_cast<std::size_t>(
464  tot_nbits % unsigned(GMP_NUMB_BITS) == 0u ?
465  tot_nbits / unsigned(GMP_NUMB_BITS) :
466  tot_nbits / unsigned(GMP_NUMB_BITS) + 1u);
467  // Overflow check when using read_uint.
468  static_assert(unsigned(GMP_LIMB_BITS) <= std::numeric_limits<std::size_t>::max() / max_n_gmp_limbs,
469  "Overflow error.");
470  for (std::size_t i = 0u; i < n_gmp_limbs; ++i) {
471  m_limbs[i] = read_uint< ::mp_limb_t,T::total_bits - T::limb_bits,
472  unsigned(GMP_LIMB_BITS - GMP_NUMB_BITS)>
473  (n.m_limbs.data(),asize,i);
474  }
475  // Fill in the missing limbs, otherwise we have indeterminate values.
476  for (std::size_t i = n_gmp_limbs; i < max_n_gmp_limbs; ++i) {
477  m_limbs[i] = 0u;
478  }
479  // Final assignment.
480  static_assert(max_n_gmp_limbs <= static_cast<std::make_unsigned<mpz_alloc_t>::type>
481  (std::numeric_limits<mpz_alloc_t>::max()),
482  "Overflow error.");
483  // There should always be some space allocated in a proper mpz struct.
484  m_mpz._mp_alloc = static_cast<mpz_alloc_t>(max_n_gmp_limbs);
485  static_assert(max_n_gmp_limbs <=
486  static_cast<std::make_unsigned<mpz_size_t>::type>(detail::safe_abs_sint<mpz_size_t>::value),
487  "Overflow error.");
488  m_mpz._mp_size = sign ? static_cast<mpz_size_t>(n_gmp_limbs) : -static_cast<mpz_size_t>(n_gmp_limbs);
489  m_mpz._mp_d = m_limbs.data();
490  }

556  // Compare absolute values of two integers whose sizes are the same in absolute value.
557  static int compare(const static_integer &a, const static_integer &b, const mpz_size_t &size)
558  {

608  // NOTE: the idea here is that it could happen that limb_bits is smaller than the actual total bits
609  // used for the representation of limb_t. In such a case, in arithmetic operations whenever we cast from

621  static int raw_add(static_integer &res, const static_integer &x, const static_integer &y)
622  {
623  piranha_assert(x.abs_size() <= 2 && y.abs_size() <= 2);
624  const dlimb_t lo = static_cast<dlimb_t>(static_cast<dlimb_t>(x.m_limbs[0u]) + y.m_limbs[0u]);
625  const dlimb_t hi = static_cast<dlimb_t>((static_cast<dlimb_t>(x.m_limbs[1u]) + y.m_limbs[1u]) + (lo >> limb_bits));
626  // NOTE: exit before modifying anything here, so that res is not modified.
627  if (unlikely(static_cast<limb_t>(hi >> limb_bits) != 0u)) {
628  return 1;
629  }
630  res.m_limbs[0u] = static_cast<limb_t>(lo);
631  res.m_limbs[1u] = static_cast<limb_t>(hi);
632  res._mp_size = res.calculate_n_limbs();
633  res.clear_extra_bits();
634  return 0;
635  }
636  static void raw_sub(static_integer &res, const static_integer &x, const static_integer &y)
637  {
638  piranha_assert(x.abs_size() <= 2 && y.abs_size() <= 2);
639  piranha_assert(x.abs_size() >= y.abs_size());
640  piranha_assert(x.m_limbs[1u] >= y.m_limbs[1u]);
641  const bool has_borrow = x.m_limbs[0u] < y.m_limbs[0u];
642  piranha_assert(x.m_limbs[1u] > y.m_limbs[1u] || !has_borrow);
643  res.m_limbs[0u] = static_cast<limb_t>(x.m_limbs[0u] - y.m_limbs[0u]);
644  res.m_limbs[1u] = static_cast<limb_t>((x.m_limbs[1u] - y.m_limbs[1u]) - limb_t(has_borrow));
645  res._mp_size = res.calculate_n_limbs();
646  res.clear_extra_bits();
647  }
648  template <bool AddOrSub>
649  static int add_or_sub(static_integer &res, const static_integer &x, const static_integer &y)
650  {
651  mpz_size_t asizex = x._mp_size, asizey = static_cast<mpz_size_t>(AddOrSub ? y._mp_size : -y._mp_size);
652  bool signx = true, signy = true;
653  if (asizex < 0) {
654  asizex = -asizex;
655  signx = false;
656  }
657  if (asizey < 0) {

665  }

667  res.negate();

669  } else {

671  raw_sub(res,x,y);

673  res.negate();
674  }
675  } else {
676  raw_sub(res,y,x);
677  if (!signy) {
678  res.negate();
679  }
680  }
681  }
682  return 0;
683  }

692  static void raw_mul(static_integer &res, const static_integer &x, const static_integer &y, const mpz_size_t &asizex,
693  const mpz_size_t &asizey)
694  {
695  piranha_assert(asizex > 0 && asizey > 0);
696  const dlimb_t lo = static_cast<dlimb_t>(static_cast<dlimb_t>(x.m_limbs[0u]) * y.m_limbs[0u]);
697  res.m_limbs[0u] = static_cast<limb_t>(lo);

701  res.clear_extra_bits();

703  }

775  {
776  mpz_size_t asizea = _mp_size, asizeb = b._mp_size, asizec = c._mp_size;
777  bool signa = true, signb = true, signc = true;
778  if (asizea < 0) {
779  asizea = -asizea;
780  signa = false;
781  }
782  if (asizeb < 0) {
783  asizeb = -asizeb;
784  signb = false;
785  }
786  if (asizec < 0) {
787  asizec = -asizec;
788  signc = false;
789  }
790  piranha_assert(asizea <= 2);
791  if (unlikely(asizeb > 1 || asizec > 1)) {
792  return 1;
793  }
794  if (unlikely(asizeb == 0 || asizec == 0)) {
795  return 0;

801  piranha_assert(asizetmp <= 2 && asizetmp > 0);
802  if (signa == signtmp) {
803  if (unlikely(raw_add(*this,*this,tmp))) {
804  return 1;
805  }
806  if (!signa) {
807  negate();
808  }
809  } else {
810  if (asizea > asizetmp || (asizea == asizetmp && compare(*this,tmp,asizea) >= 0)) {
811  raw_sub(*this,*this,tmp);
812  if (!signa) {
813  negate();
814  }
815  } else {
816  raw_sub(*this,tmp,*this);
817  if (!signtmp) {
818  negate();
819  }
820  }
821  }

824  // Left-shift by one.
825  void lshift1()
826  {
827  piranha_assert(m_limbs[1u] < (limb_t(1) << (limb_bits - 1u)));
828  using size_type = typename limbs_type::size_type;
829  // Shift both limbs.
830  dlimb_t lo = static_cast<dlimb_t>(static_cast<dlimb_t>(m_limbs[0u]) << dlimb_t(1));
831  dlimb_t hi = static_cast<dlimb_t>(static_cast<dlimb_t>(m_limbs[1u]) << dlimb_t(1));
832  m_limbs[0u] = static_cast<limb_t>(lo);
833  m_limbs[1u] = static_cast<limb_t>(hi + (lo >> limb_bits));
834  piranha_assert((hi >> limb_bits) != 1u);
835  mpz_size_t asize = _mp_size;
836  bool sign = true;
837  if (asize < 0) {
838  asize = -asize;
839  sign = false;
840  }
841  if (asize < 2) {
842  asize = static_cast<mpz_size_t>(asize + (m_limbs[static_cast<size_type>(asize)] != 0u));
843  _mp_size = static_cast<mpz_size_t>(sign ? asize : -asize);
844  }
845  clear_extra_bits();
846  }
847  // Division.
848  static void div(static_integer &q, static_integer &r, const static_integer &a, const static_integer &b)
849  {
850  piranha_assert(!b.is_zero());
851  // NOTE: here in principle q/r could overlap with a or b (e.g., in in-place division).
852  // We need to first read everything we need from a and b, and only then write into q/r.
853  // Store the signs.
854  const bool signa = a._mp_size >= 0, signb = b._mp_size >= 0;
855  // Compute the result in dlimb_t.
856  const dlimb_t ad = static_cast<dlimb_t>(a.m_limbs[0u] + (static_cast<dlimb_t>(a.m_limbs[1u]) << limb_bits)),
857  bd = static_cast<dlimb_t>(b.m_limbs[0u] + (static_cast<dlimb_t>(b.m_limbs[1u]) << limb_bits));
858  const dlimb_t qd = static_cast<dlimb_t>(ad / bd), rd = static_cast<dlimb_t>(ad % bd);
859  // Convert back to array of limb_t.
860  q.m_limbs[0u] = static_cast<limb_t>(qd);
861  q.m_limbs[1u] = static_cast<limb_t>(qd >> limb_bits);
862  q.clear_extra_bits();
863  q._mp_size = q.calculate_n_limbs();
864  r.m_limbs[0u] = static_cast<limb_t>(rd);
865  r.m_limbs[1u] = static_cast<limb_t>(rd >> limb_bits);
866  r.clear_extra_bits();
867  r._mp_size = r.calculate_n_limbs();
868  // The sign of the remainder is the same as the numerator.
869  if (!signa) {
870  r.negate();
871  }
872  // The sign of the quotient is the sign of a/b.
873  if (signa != signb) {
874  q.negate();
875  }
876  }
877  // Compute the number of bits used in the representation of the integer.
878  limb_t bits_size() const
879  {
880  using size_type = typename limbs_type::size_type;
881  const auto asize = abs_size();
882  if (asize == 0) {
883  return 0u;
884  }
885  const auto idx = static_cast<size_type>(asize - 1);
886  limb_t size = static_cast<limb_t>(limb_bits * idx), limb = m_limbs[idx];
887  while (limb != 0u) {
888  ++size;
889  limb = static_cast<limb_t>(limb >> 1u);
890  }
891  return size;
892  }
893  limb_t test_bit(const limb_t &idx) const
894  {
895  using size_type = typename limbs_type::size_type;
896  piranha_assert(idx < limb_bits * 2u);
897  const auto quot = static_cast<limb_t>(idx / limb_bits), rem = static_cast<limb_t>(idx % limb_bits);
898  return (static_cast<limb_t>(m_limbs[static_cast<size_type>(quot)] & static_cast<limb_t>(limb_t(1u) << rem)) != 0u);
899  }
900  // Some metaprogramming to make sure we can compute the hash safely. A bit of paranoid defensive programming.
901  struct hash_checks
902  {
903  // Total number of bits that can be stored. We know already this operation is safe.
904  static const limb_t tot_bits = static_cast<limb_t>(limb_bits * 2u);
905  static const unsigned nbits_size_t = static_cast<unsigned>(std::numeric_limits<std::size_t>::digits);
906  static const limb_t q = static_cast<limb_t>(tot_bits / nbits_size_t);
907  static const limb_t r = static_cast<limb_t>(tot_bits % nbits_size_t);
908  // Max number of size_t elements that can be extracted.
909  static const limb_t max_n_size_t = q + limb_t(r != 0u);
910  static const bool value =
911  // Compute tot_nbits as unsigned.
912  (tot_bits < std::numeric_limits<unsigned>::max()) &&
913  // Convert n_size_t to std::size_t for use as function param in read_uint.
914  (max_n_size_t < std::numeric_limits<std::size_t>::max()) &&
915  // Multiplication in read_uint that could overflow. max_n_size_t - 1u is the maximum
916  // value of the index param.
917  (limb_t(max_n_size_t - 1u) < std::size_t(std::numeric_limits<std::size_t>::max() / nbits_size_t));
918  };

923  if (_mp_size == 0) {
924  return 0;
925  }
926  // Use sign as seed value.
927  mpz_size_t asize = _mp_size;
928  std::size_t retval = 1u;
929  if (_mp_size < 0) {
930  asize = -asize;
931  retval = static_cast<std::size_t>(-1);
932  }
933  // Determine the number of std::size_t to extract.
934  const unsigned nbits_size_t = static_cast<unsigned>(std::numeric_limits<std::size_t>::digits),
935  tot_nbits = static_cast<unsigned>(limb_bits * static_cast<unsigned>(asize)),
936  q = tot_nbits / nbits_size_t, r = tot_nbits % nbits_size_t,
937  n_size_t = q + unsigned(r != 0u);
938  for (unsigned i = 0u; i < n_size_t; ++i) {
939  boost::hash_combine(retval,read_uint<std::size_t,total_bits - limb_bits>(&m_limbs[0u],2u,static_cast<std::size_t>(i)));
940  }
941  return retval;
942  }

1461  if (s == 0) {

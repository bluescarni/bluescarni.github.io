40 template <typename VType, typename KaType, typename T>
41 inline VType km_unpack(const symbol_set &args, const T &value)
42 {
43  if (unlikely(args.size() > VType::max_size)) {
44  piranha_throw(std::invalid_argument,"input set of arguments is too large for unpacking");
45  }
46  VType retval(static_cast<typename VType::size_type>(args.size()),0);
47  piranha_assert(args.size() == retval.size());
48  KaType::decode(retval,value);

50 }

52 template <typename VType, typename KaType, typename T>
53 inline T km_merge_args(const symbol_set &orig_args, const symbol_set &new_args, const T &value)
54 {
55  using size_type = min_int<typename VType::size_type,decltype(new_args.size())>;
56  if (unlikely(new_args.size() <= orig_args.size() ||
57  !std::includes(new_args.begin(),new_args.end(),orig_args.begin(),orig_args.end())))
58  {
59  piranha_throw(std::invalid_argument,"invalid argument(s) for symbol set merging");
60  }
61  piranha_assert(std::is_sorted(orig_args.begin(),orig_args.end()));
62  piranha_assert(std::is_sorted(new_args.begin(),new_args.end()));
63  const auto old_vector = km_unpack<VType,KaType>(orig_args,value);
64  VType new_vector;
65  auto it_new = new_args.begin();
66  for (size_type i = 0u; i < old_vector.size(); ++i, ++it_new) {
67  while (*it_new != orig_args[i]) {
68  // NOTE: for arbitrary int types, value_type(0) might throw. Update docs
69  // if needed.
70  new_vector.push_back(T(0));
71  piranha_assert(it_new != new_args.end());
72  ++it_new;
73  piranha_assert(it_new != new_args.end());
74  }
75  new_vector.push_back(old_vector[i]);
76  }
77  // Fill up arguments at the tail of new_args but not in orig_args.
78  for (; it_new != new_args.end(); ++it_new) {
79  new_vector.push_back(T(0));
80  }
81  piranha_assert(new_vector.size() == new_args.size());
82  // Return new encoded value.
83  return KaType::encode(new_vector);
84 }

86 template <typename T>
87 inline void km_safe_adder(T &a, const T &b)
88 {
89  if (b >= T(0)) {
90  if (unlikely(a > std::numeric_limits<T>::max() - b)) {
91  piranha_throw(std::overflow_error,"overflow in the addition of two exponents in a Kronecker monomial");
92  }
93  } else {
94  if (unlikely(a < std::numeric_limits<T>::min() - b)) {
95  piranha_throw(std::overflow_error,"overflow in the addition of two exponents in a Kronecker monomial");
96  }
97  }
98  a = static_cast<T>(a + b);
99 }

102 inline void km_trim_identify(symbol_set &candidates, const symbol_set &args, const T &value)
103 {
104  const VType tmp = km_unpack<VType,KaType>(args,value);
105  for (min_int<decltype(tmp.size()),decltype(args.size())> i = 0u; i < tmp.size(); ++i) {
106  if (!math::is_zero(tmp[i]) && std::binary_search(candidates.begin(),candidates.end(),args[i])) {
107  candidates.remove(args[i]);
108  }
109  }

112 template <typename VType, typename KaType, typename T>
113 inline T km_trim(const symbol_set &trim_args, const symbol_set &orig_args, const T &value)
114 {
115  using size_type = min_int<typename VType::size_type,decltype(orig_args.size())>;
116  const VType tmp = km_unpack<VType,KaType>(orig_args,value);
117  VType new_vector;
118  for (size_type i = 0u; i < tmp.size(); ++i) {
119  if (!std::binary_search(trim_args.begin(),trim_args.end(),orig_args[i])) {
120  new_vector.push_back(tmp[i]);
121  }
122  }
123  return KaType::encode(new_vector);
124 }

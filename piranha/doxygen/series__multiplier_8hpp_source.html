178  template <typename Functor>
179  Series execute() const
180  {
181  // Do not do anything if one of the two series is empty.
182  if (unlikely(m_s1->empty() || m_s2->empty())) {
183  // NOTE: requirement is ok, a series must be def-ctible.
184  return Series{};
185  }
186  // This is the size type that will be used throughout the calculations.
187  using size_type = decltype(m_v1.size());
188  const size_type size1 = m_v1.size(), size2 = m_v2.size();
189  piranha_assert(size1 && size2);
190  // Establish the number of threads to use.
191  // NOTE: this corresponds to circa 2% overhead from thread management on a common desktop
192  // machine around 2012.
193  // NOTE: tuning parameter.
194  size_type n_threads = safe_cast<size_type>(thread_pool::use_threads(
195  integer(size1) * size2,integer(500000L)
196  ));
197  piranha_assert(n_threads);
198  // An additional check on n_threads is that its size is not greater than the size of the first series,
199  // as we are using the first operand to split up the work.
200  if (n_threads > size1) {
201  n_threads = size1;
202  }
203  piranha_assert(n_threads >= 1u);
204  if (likely(n_threads == 1u)) {
205  Series retval;
206  retval.m_symbol_set = m_s1->m_symbol_set;
207  Functor f(&m_v1[0u],size1,&m_v2[0u],size2,retval);
208  const auto tmp = rehasher(f);
209  blocked_multiplication(f);
210  if (tmp.first) {
211  trace_estimates(retval.size(),tmp.second);
212  }
213  return retval;
214  } else {
215  // Build the return values and the multiplication functors.
216  std::list<Series,cache_aligning_allocator<Series>> retval_list;
217  std::list<Functor,cache_aligning_allocator<Functor>> functor_list;
218  const auto block_size = size1 / n_threads;
219  for (size_type i = 0u; i < n_threads; ++i) {
220  // Last thread needs a different size from block_size.
221  const size_type s1 = (i == n_threads - 1u) ? (size1 - i * block_size) : block_size;
222  // NOTE: ok, series must be def-ctible.
223  retval_list.emplace_back();
224  retval_list.back().m_symbol_set = m_s1->m_symbol_set;
225  functor_list.push_back(Functor(&m_v1[0u] + i * block_size,s1,&m_v2[0u],size2,retval_list.back()));
226  }
227  auto f_it = functor_list.begin();
228  auto r_it = retval_list.begin();
229  future_list<std::future<void>> f_list;
230  try {
231  for (size_type i = 0u; i < n_threads; ++i, ++f_it, ++r_it) {
232  // Functor for use in the thread.
233  auto f = [f_it,r_it,this]() {
234  const auto tmp = this->rehasher(*f_it);
235  this->blocked_multiplication(*f_it);
236  if (tmp.first) {
237  this->trace_estimates(r_it->m_container.size(),tmp.second);
238  }
239  };
240  f_list.push_back(thread_pool::enqueue(static_cast<unsigned>(i),f));
241  }
242  f_list.wait_all();
243  f_list.get_all();
244  } catch (...) {
245  f_list.wait_all();
246  throw;
247  }
248  Series retval;
249  retval.m_symbol_set = m_s1->m_symbol_set;
250  auto final_estimate = estimate_final_series_size(Functor(&m_v1[0u],size1,&m_v2[0u],size2,retval));
251  // We want to make sure that final_estimate contains at least 1 element, so that we can use faster low-level
252  // methods in hash_set.
253  if (unlikely(!final_estimate)) {
254  final_estimate = 1u;
255  }
256  // Try to see if a series already has enough buckets.
257  auto it = std::find_if(retval_list.begin(),retval_list.end(),[&final_estimate](const Series &r) {
258  return static_cast<double>(r.m_container.bucket_count()) * r.m_container.max_load_factor() >= final_estimate;
259  });
260  if (it != retval_list.end()) {
261  retval = std::move(*it);
262  retval_list.erase(it);
263  } else {
264  // Otherwise, just rehash to the desired value, corrected for max load factor.
265  retval.m_container.rehash(
266  safe_cast<typename Series::size_type>(std::ceil(static_cast<double>(final_estimate) / retval.m_container.max_load_factor()))
267  );
268  }
269  // Cleanup functor that will erase all elements in retval_list.
270  auto cleanup = [&retval_list]() {
271  std::for_each(retval_list.begin(),retval_list.end(),[](Series &r) {r.m_container.clear();});
272  };
273  try {
274  final_merge(retval,retval_list,n_threads);
275  // Here the estimate will be correct if the initial estimate (corrected for load factor)
276  // is at least equal to the final real bucket count.
277  trace_estimates(retval.m_container.size(),final_estimate);
278  } catch (...) {
279  // Do the cleanup before re-throwing, as we use mutable iterators in final_merge.
280  cleanup();
281  // Clear also retval: it could have bogus number of elements if errors arise in the final merge.
282  retval.m_container.clear();
283  throw;
284  }
285  // Clean up the temporary series.
286  cleanup();
287  return retval;
288  }

381  template <typename Functor>
382  static void blocked_multiplication(const Functor &f)
383  {
384  typedef typename std::decay<decltype(f.m_s1)>::type size_type;
385  // NOTE: hard-coded block size of 256.
386  static_assert(std::numeric_limits<size_type>::max() >= 256u, "Invalid size type.");
387  const size_type size1 = f.m_s1, size2 = f.m_s2, bsize1 = 256u, nblocks1 = size1 / bsize1, bsize2 = bsize1, nblocks2 = size2 / bsize2;
388  // Start and end of last (possibly irregular) blocks.
389  const size_type i_ir_start = nblocks1 * bsize1, i_ir_end = size1;
390  const size_type j_ir_start = nblocks2 * bsize2, j_ir_end = size2;
391  for (size_type n1 = 0u; n1 < nblocks1; ++n1) {
392  const size_type i_start = n1 * bsize1, i_end = i_start + bsize1;
393  // regulars1 * regulars2
394  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
395  const size_type j_start = n2 * bsize2, j_end = j_start + bsize2;
396  for (size_type i = i_start; i < i_end; ++i) {
397  for (size_type j = j_start; j < j_end; ++j) {
398  f(i,j);
399  f.insert();
400  }
401  }
402  }
403  // regulars1 * rem2
404  for (size_type i = i_start; i < i_end; ++i) {
405  for (size_type j = j_ir_start; j < j_ir_end; ++j) {
406  f(i,j);
407  f.insert();
408  }
409  }
410  }
411  // rem1 * regulars2
412  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
413  const size_type j_start = n2 * bsize2, j_end = j_start + bsize2;
414  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
415  for (size_type j = j_start; j < j_end; ++j) {
416  f(i,j);
417  f.insert();
418  }
419  }
420  }
421  // rem1 * rem2.
422  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
423  for (size_type j = j_ir_start; j < j_ir_end; ++j) {
424  f(i,j);
425  f.insert();
426  }
427  }

446  template <typename Functor>
447  static typename Series::size_type estimate_final_series_size(const Functor &f)
448  {
449  typedef typename Series::size_type bucket_size_type;
450  typedef typename std::decay<decltype(f.m_s1)>::type size_type;
451  const size_type size1 = f.m_s1, size2 = f.m_s2;
452  // If one of the two series is empty, just return 0.
453  if (unlikely(!size1 || !size2)) {
454  return 0u;
455  }
456  // Cache reference to return series.
457  auto &retval = f.m_retval;
458  // NOTE: Hard-coded number of trials = 10.
459  // NOTE: here consider that in case of extremely sparse series with few terms (e.g., next to the lower limit
460  // for which this function is called) this will incur in noticeable overhead, since we will need many term-by-term
461  // before encountering the first duplicate.
462  const auto ntrials = 10u;
463  // NOTE: Hard-coded value for the estimation multiplier.
464  // NOTE: This value should be tuned for performance/memory usage tradeoffs.
465  const auto multiplier = 2;
466  // Size of the multiplication result
467  // Vectors of indices.
468  std::vector<size_type> v_idx1, v_idx2;
469  for (size_type i = 0u; i < size1; ++i) {
470  v_idx1.push_back(i);
471  }
472  for (size_type i = 0u; i < size2; ++i) {
473  v_idx2.push_back(i);
474  }
475  // Maxium number of random multiplications before which a duplicate term must be generated.
476  const size_type max_M = static_cast<size_type>(((integer(size1) * size2) / multiplier).sqrt());
477  // Random number engine.
478  std::mt19937 engine;
479  // Init counter.
480  integer total(0);
481  // Go with the trials.
482  // NOTE: This could be easily parallelised, but not sure if it is worth.
483  for (auto n = 0u; n < ntrials; ++n) {
484  // Randomise.
485  std::shuffle(v_idx1.begin(),v_idx1.end(),engine);
486  std::shuffle(v_idx2.begin(),v_idx2.end(),engine);
487  size_type count = 0u;
488  auto it1 = v_idx1.begin(), it2 = v_idx2.begin();
489  while (count < max_M) {
490  if (it1 == v_idx1.end()) {
491  // Each time we wrap around the first series,
492  // wrap around also the second one and rotate it.
493  it1 = v_idx1.begin();
494  auto middle = v_idx2.end();
495  --middle;
496  std::rotate(v_idx2.begin(),middle,v_idx2.end());
497  it2 = v_idx2.begin();
498  }
499  if (it2 == v_idx2.end()) {
500  it2 = v_idx2.begin();
501  }
502  // Perform multiplication and check if it produces a new term.
503  f(*it1,*it2);
504  f.insert();
505  constexpr std::size_t result_size = Series::term_type::key_type::multiply_arity;
506  // Check for unlikely overflow when increasing count.
507  if (unlikely(result_size > std::numeric_limits<size_type>::max() ||
508  count > std::numeric_limits<size_type>::max() - result_size))
509  {
510  piranha_throw(std::overflow_error,"overflow error");
511  }
512  if (retval.size() != count + result_size) {
513  break;
514  }
515  // Increase cycle variables.
516  count += result_size;
517  ++it1;
518  ++it2;
519  }
520  total += count;
521  // Reset retval.
522  retval.m_container.clear();
523  }
524  const auto mean = total / ntrials;
525  // Avoid division by zero.
526  if (total.sign()) {
527  const integer M = (mean * mean * multiplier * total) / total;
528  return static_cast<bucket_size_type>(M);
529  } else {
530  return static_cast<bucket_size_type>(mean * mean * multiplier);
531  }
532  }

552  static void trace_estimates(const typename Series::size_type &real_size, const typename Series::size_type &estimate)
553  {
554  if (unlikely(!real_size)) {
555  return;
556  }
557  tracing::trace("number_of_estimates",[](boost::any &x) {
558  if (unlikely(x.empty())) {
559  x = 0ull;
560  }
561  auto ptr = boost::any_cast<unsigned long long>(&x);
562  if (likely((bool)ptr)) {
563  ++*ptr;
564  }
565  });
566  tracing::trace("number_of_correct_estimates",[real_size,estimate](boost::any &x) {
567  if (unlikely(x.empty())) {
568  x = 0ull;
569  }
570  auto ptr = boost::any_cast<unsigned long long>(&x);
571  if (likely((bool)ptr)) {
572  *ptr += static_cast<unsigned long long>(estimate >= real_size);
573  }
574  });
575  tracing::trace("accumulated_estimate_ratio",[real_size,estimate](boost::any &x) {
576  if (unlikely(x.empty())) {
577  x = 0.;
578  }
579  auto ptr = boost::any_cast<double>(&x);
580  if (likely((bool)ptr && estimate)) {
581  *ptr += static_cast<double>(estimate) / static_cast<double>(real_size);
582  }
583  });

586  template <typename RetvalList, typename Size>
587  static void final_merge(Series &retval, RetvalList &retval_list, const Size &n_threads)
588  {
589  piranha_assert(n_threads > 1u);
590  piranha_assert(retval.m_container.bucket_count());
591  typedef typename std::vector<std::pair<typename Series::size_type,decltype(retval.m_container._m_begin())>> container_type;
592  typedef typename container_type::size_type size_type;
593  // First, let's fill a vector assigning each term of each element in retval_list to a bucket in retval.
594  const size_type size = static_cast<size_type>(
595  std::accumulate(retval_list.begin(),retval_list.end(),integer(0),[](const integer &n, const Series &r) {return n + r.size();}));
596  container_type idx(size);
597  future_list<std::future<void>> f_list1;
598  size_type i = 0u;
599  piranha_assert(retval_list.size() <= n_threads);
600  try {
601  unsigned thread_idx = 0u;
602  for (auto r_it = retval_list.begin(); r_it != retval_list.end(); ++r_it, ++thread_idx) {
603  auto f = [&idx,&retval,i,r_it]() {
604  const auto it_f = r_it->m_container._m_end();
605  // NOTE: size_type can represent the sum of the sizes of all retvals,
606  // so it will not overflow here.
607  size_type tmp_i = i;
608  for (auto it = r_it->m_container._m_begin(); it != it_f; ++it, ++tmp_i) {
609  piranha_assert(tmp_i < idx.size());
610  idx[tmp_i] = std::make_pair(retval.m_container._bucket(*it),it);
611  }
612  };
613  f_list1.push_back(thread_pool::enqueue(thread_idx,f));
614  i += r_it->size();
615  }
616  f_list1.wait_all();
617  f_list1.get_all();
618  } catch (...) {
619  f_list1.wait_all();
620  throw;
621  }
622  future_list<std::future<void>> f_list2;
623  std::mutex m;
624  std::vector<integer> new_sizes;
625  new_sizes.reserve(static_cast<std::vector<integer>::size_type>(n_threads));
626  if (unlikely(new_sizes.capacity() != n_threads)) {
627  piranha_throw(std::bad_alloc,);
628  }
629  const typename Series::size_type bucket_count = retval.m_container.bucket_count(), block_size = bucket_count / n_threads;
630  try {
631  for (Size n = 0u; n < n_threads; ++n) {
632  // These are the bucket indices allowed to the current thread.
633  const typename Series::size_type start = n * block_size, end = (n == n_threads - 1u) ? bucket_count : (n + 1u) * block_size;
634  auto f = [start,end,&size,&retval,&idx,&m,&new_sizes]() {
635  size_type count_plus = 0u, count_minus = 0u;
636  for (size_type i = 0u; i < size; ++i) {
637  const auto &bucket_idx = idx[i].first;
638  auto &term_it = idx[i].second;
639  if (bucket_idx >= start && bucket_idx < end) {
640  // Reconstruct part of series insertion, without the update in the number of elements.
641  // NOTE: compatibility and ignorability do not matter, as terms being inserted come from series
642  // anyway. Same for check on bucket_count.
643  const auto it = retval.m_container._find(*term_it,bucket_idx);
644  if (it == retval.m_container.end()) {
645  // Term is new.
646  retval.m_container._unique_insert(std::move(*term_it),bucket_idx);
647  // Update term count.
648  piranha_assert(count_plus < std::numeric_limits<size_type>::max());
649  ++count_plus;
650  } else {
651  // Assert the existing term is not ignorable and it is compatible.
652  piranha_assert(!it->is_ignorable(retval.m_symbol_set) && it->is_compatible(retval.m_symbol_set));
653  // Cleanup function.
654  auto cleanup = [&]() {
655  if (unlikely(it->is_ignorable(retval.m_symbol_set))) {
656  retval.m_container._erase(it);
657  // After term is erased, update count.
658  piranha_assert(count_minus < std::numeric_limits<size_type>::max());
659  ++count_minus;
660  }
661  };
662  try {
663  // The term exists already, update it.
664  retval.template insertion_cf_arithmetics<true>(it,std::move(*term_it));
665  // Check if the term has become ignorable after the modification.
666  cleanup();
667  } catch (...) {
668  cleanup();
669  throw;
670  }
671  }
672  }
673  }
674  // Store the new size.
675  std::lock_guard<std::mutex> lock(m);
676  new_sizes.push_back(integer(count_plus) - integer(count_minus));
677  };
678  // NOTE: here the static cast is safe as nthreads is coming from a call to
679  // the size of the thread pool.
680  f_list2.push_back(thread_pool::enqueue(static_cast<unsigned>(n),f));
681  }
682  f_list2.wait_all();
683  f_list2.get_all();
684  } catch (...) {
685  f_list2.wait_all();
686  throw;
687  }
688  // Final update of retval's size.
689  const auto new_size = std::accumulate(new_sizes.begin(),new_sizes.end(),integer(0));
690  piranha_assert(new_size + retval.m_container.size() >= 0);

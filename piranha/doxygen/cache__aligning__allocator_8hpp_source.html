43 template <typename T>
44 class cache_aligning_allocator: public dynamic_aligning_allocator<T>
45 {
46  using base = dynamic_aligning_allocator<T>;
47  static std::size_t determine_alignment()
48  {
49 #if !defined(PIRANHA_HAVE_MEMORY_ALIGNMENT_PRIMITIVES)
50  return 0u;
51 #endif
52  try {
53  const std::size_t alignment = boost::numeric_cast<std::size_t>(settings::get_cache_line_size());
54  if (!alignment_check<T>(alignment)) {
55  return 0u;
56  }
57  return alignment;
58  } catch (...) {
59  return 0u;
60  }
61  }

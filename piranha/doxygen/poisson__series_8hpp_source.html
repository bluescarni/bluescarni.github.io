159  // Sin/cos overrides implementation.
160  // If cf is a suitable polynomial, attempt the symbolic sin/cos with the integral linear combination.
161  template <bool IsCos, typename T, typename std::enable_if<cf_poly_has_icast<T>::value,int>::type = 0>
162  poisson_series sin_cos_impl() const
163  {
164  // Do something only if the series is equivalent to a polynomial.
165  if (this->is_single_coefficient() && !this->empty()) {
166  try {
167  // Shortcuts.
168  typedef term<Cf,rtk_monomial> term_type;
169  typedef typename term_type::key_type key_type;
170  typedef typename key_type::value_type value_type;
171  // Try to get the integral combination.
172  auto lc = this->m_container.begin()->m_cf.integral_combination();
173  // Change sign if needed.
174  bool sign_change = false;
175  if (!lc.empty() && lc.begin()->second.sign() < 0) {
176  std::for_each(lc.begin(),lc.end(),[](std::pair<const std::string,integer> &p) {p.second.negate();});
177  sign_change = true;
178  }
179  // Return value.
180  poisson_series retval;
181  // Build vector of integral multipliers.
182  std::vector<value_type> v;
183  for (auto it = lc.begin(); it != lc.end(); ++it) {
184  retval.m_symbol_set.add(it->first);
185  // NOTE: this should probably be a safe_cast, if we ever implement it.
186  // The value type here could be anything, and not guaranteed to be castable,
187  // even if in the current implementation this is guaranteed to be a signed
188  // int of some kind.
189  v.push_back(static_cast<value_type>(it->second));
190  }
191  // Build term, fix signs and flavour and move-insert it.
192  term_type term(Cf(1),key_type(v.begin(),v.end()));
193  if (!IsCos) {
194  term.m_key.set_flavour(false);
195  if (sign_change) {
196  math::negate(term.m_cf);
197  }
198  }
199  retval.insert(std::move(term));
200  return retval;
201  } catch (const std::invalid_argument &) {
202  // Interpret invalid_argument as a failure in extracting integral combination,
203  // and move on.
204  }
205  }
206  return sin_cos_cf_impl<IsCos,T>();
207  }
208  // If cf is not a suitable poly, go for the standard implementation.

210  poisson_series sin_cos_impl() const

244  poisson_series integrate_impl(const symbol &s, const typename base::term_type &term,
245  const std::true_type &) const
246  {
247  typedef typename base::term_type term_type;
248  typedef typename term_type::cf_type cf_type;
249  integer degree;
250  try {
251  degree = safe_cast<integer>(math::degree(term.m_cf,{s.get_name()}));
252  } catch (const std::invalid_argument &) {
253  piranha_throw(std::invalid_argument,
254  "unable to perform Poisson series integration: cannot extract the integral form of a polynomial degree");
255  }
256  // If the variable is in both cf and key, and the cf degree is negative, we cannot integrate.
257  if (degree.sign() < 0) {
258  piranha_throw(std::invalid_argument,
259  "unable to perform Poisson series integration: polynomial coefficient has negative integral degree");
260  }
261  // Init retval and auxiliary quantities for the iteration.
262  poisson_series retval;
263  retval.m_symbol_set = this->m_symbol_set;
264  auto key_int = term.m_key.integrate(s,this->m_symbol_set);
265  // NOTE: here we are sure that the variable is contained in the monomial.
266  piranha_assert(key_int.first.sign() != 0);
267  cf_type p_cf(term.m_cf / key_int.first);
268  retval.insert(term_type(p_cf,key_int.second));
269  for (integer i(1); i <= degree; ++i) {
270  key_int = key_int.second.integrate(s,this->m_symbol_set);
271  piranha_assert(key_int.first.sign() != 0);
272  p_cf = math::partial(p_cf / key_int.first,s.get_name());
273  // Sign change due to the second portion of integration by part.
274  math::negate(p_cf);
275  retval.insert(term_type(p_cf,key_int.second));
276  }
277  return retval;
278  }

291  template <typename T>

293  piranha::poisson_series<divisor_series<t_int_div_cf_type<T>,t_int_div_key_type<T>>>>::type;
294  // Overload if cf is not a divisor series already. The result will be a Poisson series with the same key type, in which the coefficient
295  // is a divisor series whose coefficient is calculated from the operations needed in the integration, and the key type is a divisor whose
296  // value type is deduced from the trigonometric key.
297  template <typename T = poisson_series, typename std::enable_if<!detail::is_divisor_series<typename T::term_type::cf_type>::value,int>::type = 0>
298  ti_type_<T> t_integrate_impl() const
299  {
300  using return_type = ti_type_<T>;
301  // The value type of the trigonometric key.
302  using k_value_type = typename base::term_type::key_type::value_type;
303  // The divisor type in the return type.
304  using div_type = typename return_type::term_type::cf_type::term_type::key_type;
305  // Initialise the return value. It has the same set of trig arguments as this.
306  return_type retval;
307  retval.set_symbol_set(this->m_symbol_set);
308  // The symbol set for the divisor series - built from the trigonometric arguments.
309  symbol_set div_symbols;
310  for (auto it = retval.get_symbol_set().begin(); it != retval.get_symbol_set().end(); ++it) {
311  div_symbols.add(std::string("\\nu_{") + it->get_name() + "}");
312  }
313  // A temp vector of integers used to normalise the divisors coming
314  // out of the integration operation from the trig keys.
315  std::vector<integer> tmp_int;
316  // Build the return value.
317  const auto it_f = this->m_container.end();
318  for (auto it = this->m_container.begin(); it != it_f; ++it) {
319  // Clear the tmp integer vector.
320  tmp_int.clear();
321  // Get the vector of trigonometric multipliers.
322  const auto trig_vector = it->m_key.unpack(this->m_symbol_set);
323  // Copy it over to the tmp_int as integer values.
324  std::transform(trig_vector.begin(),trig_vector.end(),std::back_inserter(tmp_int),
325  [](const k_value_type &n) {return integer(n);});
326  // Determine the common divisor and if a sign flip is needed.
327  integer cd(0);
328  bool need_sign_flip = false, first_nonzero_found = false;
329  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
330  cd = detail::gcd(cd,*it2);
331  if (!first_nonzero_found && !math::is_zero(*it2)) {
332  if (*it2 < 0) {
333  need_sign_flip = true;
334  }
335  first_nonzero_found = true;
336  }
337  }
338  if (unlikely(math::is_zero(cd))) {
339  piranha_throw(std::invalid_argument,"an invalid trigonometric term was encountered while "
340  "attempting a time integration");
341  }
342  // Take the abs of the cd.
343  cd = cd.abs();
344  // Divide the vector by the common divisor, and flip the sign if needed.
345  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
346  *it2 /= cd;
347  if (need_sign_flip) {
348  it2->negate();
349  }
350  }
351  // Build first the divisor series - the coefficient of the term to be inserted
352  // into retval.
353  typename return_type::term_type::cf_type div_series;
354  // Set the arguments of the divisor series.
355  div_series.set_symbol_set(div_symbols);
356  // The coefficient of the only term of the divisor series is the original coefficient
357  // multiplied by any sign change from the integration or the change in sign in the divisors,
358  // and divided by the common divisor (cast to the appropriate type).
359  // NOTE: probably the sign flip is never needed, as the canonical form of the trigonometric keys
360  // also enforces a first nonzero multiplier.
361  typename return_type::term_type::cf_type::term_type::cf_type div_cf = (it->m_cf *
362  ((it->m_key.get_flavour() ? 1 : -1) * (need_sign_flip ? -1 : 1))) /
363  static_cast<typename div_type::value_type>(cd);
364  // Build the divisor.
365  typename div_type::value_type exponent(1);
366  typename return_type::term_type::cf_type::term_type::key_type div_key;
367  div_key.insert(tmp_int.begin(),tmp_int.end(),exponent);
368  // Insert the term into the divisor series.
369  div_series.insert(typename return_type::term_type::cf_type::term_type{std::move(div_cf),std::move(div_key)});
370  // Insert into the return value.
371  auto tmp_key = it->m_key;
372  // Switch the flavour for integration.
373  tmp_key.set_flavour(!tmp_key.get_flavour());
374  retval.insert(typename return_type::term_type{std::move(div_series),std::move(tmp_key)});
375  }
376  return retval;
377  }

501  PIRANHA_TT_CHECK(is_differentiable,cf_type);
502  // Turn name into symbol.
503  const symbol s(name);
504  poisson_series retval;
505  retval.m_symbol_set = this->m_symbol_set;
506  const auto it_f = this->m_container.end();
507  for (auto it = this->m_container.begin(); it != it_f; ++it) {
508  // Try to integrate the key first.
509  const auto key_int = it->m_key.integrate(s,this->m_symbol_set);
510  if (key_int.first.sign() == 0) {
511  // The variable does not appear in the monomial, try deferring the integration
512  // to the coefficient.
513  retval.insert(term_type(cf_type(integrate_cf(it->m_cf,name)),it->m_key));
514  continue;
515  }
516  // The variable is in the monomial, let's check if the variable is also in the coefficient.
517  if (math::is_zero(math::partial(it->m_cf,name))) {
518  // No variable in the coefficient, proceed with the integrated key and divide by multiplier.
519  retval.insert(term_type(it->m_cf / key_int.first,std::move(key_int.second)));
520  } else {
521  // With the variable both in the coefficient and the key, we only know how to proceed with polynomials.
522  retval += integrate_impl(s,*it,std::integral_constant<bool,std::is_base_of<detail::polynomial_tag,Cf>::value>());
523  }
524  }

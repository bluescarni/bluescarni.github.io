166  // Integration with integrable coefficient.
167  polynomial integrate_impl(const symbol &s, const typename base::term_type &term,
168  const std::true_type &) const
169  {
170  typedef typename base::term_type term_type;
171  typedef typename term_type::cf_type cf_type;
172  typedef typename term_type::key_type key_type;
173  // Get the partial degree of the monomial in integral form.
174  integer degree;
175  const symbol_set::positions pos(this->m_symbol_set,symbol_set{s});
176  try {
177  degree = safe_cast<integer>(term.m_key.degree(pos,this->m_symbol_set));
178  } catch (const std::invalid_argument &) {
179  piranha_throw(std::invalid_argument,
180  "unable to perform polynomial integration: cannot extract the integral form of an exponent");
181  }
182  // If the degree is negative, integration by parts won't terminate.
183  if (degree.sign() < 0) {
184  piranha_throw(std::invalid_argument,
185  "unable to perform polynomial integration: negative integral exponent");
186  }
187  // Initialise retval (this is also the final retval in case the degree of s is zero).
188  polynomial retval;
189  retval.m_symbol_set = this->m_symbol_set;
190  key_type tmp_key = term.m_key;
191  cf_type i_cf(math::integrate(term.m_cf,s.get_name()));
192  retval.insert(term_type(i_cf,tmp_key));
193  for (integer i(1); i <= degree; ++i) {
194  // Update coefficient and key.
195  i_cf = cf_type(math::integrate(i_cf,s.get_name()));
196  auto partial_key = tmp_key.partial(pos,this->m_symbol_set);
197  i_cf *= std::move(partial_key.first);
198  // Account for (-1)**i.
199  math::negate(i_cf);
200  tmp_key = std::move(partial_key.second);
201  retval.insert(term_type(i_cf,tmp_key));
202  }
203  return retval;
204  }

365  polynomial integrate(const std::string &name) const
366  {
367  typedef typename base::term_type term_type;
368  typedef typename term_type::cf_type cf_type;
369  PIRANHA_TT_CHECK(is_differentiable,cf_type);
370  // Turn name into symbol.
371  const symbol s(name);
372  polynomial retval;
373  const auto it_f = this->m_container.end();
374  for (auto it = this->m_container.begin(); it != it_f; ++it) {
375  // If the derivative of the coefficient is null, we just need to deal with
376  // the integration of the key.
377  if (math::is_zero(math::partial(it->m_cf,name))) {
378  polynomial tmp;
379  symbol_set sset = this->m_symbol_set;
380  // If the variable does not exist in the arguments set, add it.
381  if (!std::binary_search(sset.begin(),sset.end(),s)) {
382  sset.add(s);
383  }
384  tmp.m_symbol_set = sset;
385  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
386  tmp.insert(term_type(it->m_cf,std::move(key_int.second)));
387  tmp /= key_int.first;
388  retval += std::move(tmp);
389  } else {
390  retval += integrate_impl(s,*it,std::integral_constant<bool,is_integrable<cf_type>::value>());
391  }
392  }
393  return retval;
394  }

489  explicit series_multiplier(const Series &s1, const Series &s2):base(s1,s2)
490  {
491  if (unlikely(this->m_s1->empty() || this->m_s2->empty())) {
492  return;
493  }
494  // NOTE: here we are sure about this since the symbol set in a series should never
495  // overflow the size of the limits, as the check for compatibility in Kronecker monomial
496  // would kick in.
497  piranha_assert(this->m_s1->m_symbol_set.size() < ka::get_limits().size());
498  piranha_assert(this->m_s1->m_symbol_set == this->m_s2->m_symbol_set);
499  const auto &limits = ka::get_limits()[this->m_s1->m_symbol_set.size()];
500  // NOTE: We need to check that the exponents of the monomials in the result do not
501  // go outside the bounds of the Kronecker codification. We need to unpack all monomials
502  // in the operands and examine them, we cannot operate on the codes for this.
503  auto it1 = this->m_v1.begin();
504  auto it2 = this->m_v2.begin();
505  // Initialise minmax values.
506  std::vector<std::pair<value_type,value_type>> minmax_values1;
507  std::vector<std::pair<value_type,value_type>> minmax_values2;
508  auto tmp_vec1 = (*it1)->m_key.unpack(this->m_s1->m_symbol_set);
509  auto tmp_vec2 = (*it2)->m_key.unpack(this->m_s1->m_symbol_set);
510  // Bounds of the Kronecker representation for each component.
511  const auto &minmax_vec = std::get<0u>(limits);
512  std::transform(tmp_vec1.begin(),tmp_vec1.end(),std::back_inserter(minmax_values1),[](const value_type &v) {
513  return std::make_pair(v,v);
514  });
515  std::transform(tmp_vec2.begin(),tmp_vec2.end(),std::back_inserter(minmax_values2),[](const value_type &v) {
516  return std::make_pair(v,v);
517  });
518  // Find the minmaxs.
519  for (; it1 != this->m_v1.end(); ++it1) {
520  tmp_vec1 = (*it1)->m_key.unpack(this->m_s1->m_symbol_set);
521  piranha_assert(tmp_vec1.size() == minmax_values1.size());
522  std::transform(minmax_values1.begin(),minmax_values1.end(),tmp_vec1.begin(),minmax_values1.begin(),
523  [](const std::pair<value_type,value_type> &p, const value_type &v) {
524  return std::make_pair(
525  v < p.first ? v : p.first,
526  v > p.second ? v : p.second
527  );
528  });
529  }
530  for (; it2 != this->m_v2.end(); ++it2) {
531  tmp_vec2 = (*it2)->m_key.unpack(this->m_s2->m_symbol_set);
532  piranha_assert(tmp_vec2.size() == minmax_values2.size());
533  std::transform(minmax_values2.begin(),minmax_values2.end(),tmp_vec2.begin(),minmax_values2.begin(),
534  [](const std::pair<value_type,value_type> &p, const value_type &v) {
535  return std::make_pair(
536  v < p.first ? v : p.first,
537  v > p.second ? v : p.second
538  );
539  });
540  }
541  // Compute the sum of the two minmaxs, using multiprecision to avoid overflow.
542  // NOTE: first store in m_minmax_values the ranges of the result only, below we will update this with the range
543  // of the operands.
544  std::transform(minmax_values1.begin(),minmax_values1.end(),minmax_values2.begin(),
545  std::back_inserter(m_minmax_values),[](const std::pair<value_type,value_type> &p1,
546  const std::pair<value_type,value_type> &p2) {
547  return std::make_pair(integer(p1.first) + integer(p2.first),integer(p1.second) + integer(p2.second));
548  });
549  piranha_assert(m_minmax_values.size() == minmax_vec.size());
550  piranha_assert(m_minmax_values.size() == minmax_values1.size());
551  piranha_assert(m_minmax_values.size() == minmax_values2.size());
552  for (decltype(m_minmax_values.size()) i = 0u; i < m_minmax_values.size(); ++i) {
553  if (unlikely(m_minmax_values[i].first < -minmax_vec[i] || m_minmax_values[i].second > minmax_vec[i])) {
554  piranha_throw(std::overflow_error,"Kronecker monomial components are out of bounds");
555  }
556  // Update with the ranges of the operands.
557  m_minmax_values[i] = std::minmax({m_minmax_values[i].first,
558  integer(minmax_values1[i].first),integer(minmax_values2[i].first),m_minmax_values[i].second,
559  integer(minmax_values1[i].second),integer(minmax_values2[i].second)});
560  }
561  }

601  // Create task from indices i in first series, j in second series (semi-open intervals).
602  task_type task_from_indices(const index_type &i_start, const index_type &i_end,
603  const index_type &j_start, const index_type &j_end, const Series &retval) const
604  {
605  const auto &v1 = this->m_v1;
606  const auto &v2 = this->m_v2;
607  piranha_assert(i_start < i_end && j_start < j_end);
608  piranha_assert(i_end <= v1.size() && j_end <= v2.size());
609  const auto b_count = retval.m_container.bucket_count();
610  piranha_assert(b_count > 0u);
611  region_type r1{0u,0u}, r2{0u,0u};
612  bool second_region = false;
613  // Addition is safe because of the limits on the max bucket count of hash_set.
614  const auto a = retval.m_container._bucket_from_hash(v1[i_start]->hash()) +
615  retval.m_container._bucket_from_hash(v2[j_start]->hash()),
616  // NOTE: we are sure that the tasks are not empty, so the -1 is safe.
617  b = retval.m_container._bucket_from_hash(v1[i_end - 1u]->hash()) +
618  retval.m_container._bucket_from_hash(v2[j_end - 1u]->hash());
619  // NOTE: using <= here as [a,b] is now a closed interval.
620  piranha_assert(a <= b);
621  piranha_assert(b <= b_count * 2u - 2u);
622  if (b < b_count || a >= b_count) {
623  r1.first = a % b_count;
624  r1.second = b % b_count;
625  } else {
626  second_region = true;
627  r1.first = a;
628  r1.second = b_count - 1u;
629  r2.first = 0u;
630  r2.second = b % b_count;
631  }
632  // Correct the case in which a second region reaches the first one (thus
633  // covering the whole range of buckets).
634  if (second_region && r2.second >= r1.first) {
635  second_region = false;
636  r1.first = 0u;
637  r1.second = b_count - 1u;
638  r2.first = 0u;
639  r2.second = 0u;
640  }
641  piranha_assert(r1.first <= r1.second);
642  piranha_assert(!second_region || (r2.first <= r2.second && r2.second < r1.first));
643  return task_type{std::make_pair(i_start,i_end),std::make_pair(j_start,j_end),r1,r2,second_region};
644  }
645  // Functor to check if region r does not overlap any of the busy ones.
646  template <typename RegionSet>
647  static bool region_checker(const region_type &r, const RegionSet &busy_regions)
648  {
649  if (busy_regions.empty()) {
650  return true;
651  }
652  // NOTE: lower bound means that it_b->first >= r.first.
653  auto it_b = busy_regions.lower_bound(r);
654  // Handle end().
655  if (it_b == busy_regions.end()) {
656  // NOTE: safe because busy_regions is not empty.
657  --it_b;
658  // Now check that the region in it_b does not overlap with
659  // the current one.
660  piranha_assert(it_b->first < r.first);
661  return it_b->second < r.first;
662  }
663  if (r.second >= it_b->first) {
664  // The end point of r1 overlaps the start point of *it_b, no good.
665  return false;
666  }
667  // Lastly, we have to check that the previous element (if any) does not
668  // overlap r.first.
669  if (it_b != busy_regions.begin()) {
670  --it_b;
671  if (it_b->second >= r.first) {
672  return false;
673  }
674  }
675  return true;
676  }
677  // Check that a set of busy regions is well-formed, for debug purposes.
678  template <typename RegionSet>

680  {

682  return true;

691  }

694  }

709  {
710  piranha_assert(m_retval.m_container.bucket_count());
711  // NOTE: here we are sure there is no overflow as the max size of the hash table is 2 ** (n - 1), hence the highest
712  // possible bucket is 2 ** (n - 1) - 1, so the highest value of the sum will be 2 ** n - 2. But the range of the size type
713  // is [0, 2 ** n - 1], so it is safe.
714  // NOTE: might use faster mod calculation here, but probably not worth it.
715  // NOTE: because tasks cannot contain empty intervals, the start of each interval will be a valid index (i.e., not end())
716  // in the term pointers vectors.
717  piranha_assert(t1.m_b1.first < m_v1.size() && t2.m_b1.first < m_v1.size() &&
718  t1.m_b2.first < m_v2.size() && t2.m_b2.first < m_v2.size() &&
719  t1.m_b1.first < t1.m_b1.second && t1.m_b2.first < t1.m_b2.second &&
720  t2.m_b1.first < t2.m_b1.second && t2.m_b2.first < t2.m_b2.second
721  );
722  return (m_retval.m_container._bucket_from_hash(m_v1[t1.m_b1.first]->hash()) +
723  m_retval.m_container._bucket_from_hash(m_v2[t1.m_b2.first]->hash())) %
724  m_retval.m_container.bucket_count() <
725  (m_retval.m_container._bucket_from_hash(m_v1[t2.m_b1.first]->hash()) +
726  m_retval.m_container._bucket_from_hash(m_v2[t2.m_b2.first]->hash())) %
727  m_retval.m_container.bucket_count();
728  }
729  const Series &m_retval;
730  const std::vector<term_type const *> &m_v1;
731  const std::vector<term_type const *> &m_v2;
732  };
733  Series execute() const
734  {
735  const index_type size1 = this->m_v1.size(), size2 = this->m_v2.size();
736  // Do not do anything if one of the two series is empty, just return an empty series.
737  if (unlikely(!size1 || !size2)) {
738  return Series{};
739  }
740  // This check is done here to avoid controlling the number of elements of the output series
741  // at every iteration of the functor.
742  const auto max_size = integer(size1) * size2;
743  if (unlikely(max_size > std::numeric_limits<bucket_size_type>::max())) {
744  piranha_throw(std::overflow_error,"possible overflow in series size");
745  }
746  // First, let's get the estimation on the size of the final series.
747  Series retval;
748  retval.m_symbol_set = this->m_s1->m_symbol_set;
749  typename Series::size_type estimate;
750  // Use the sparse functor for the estimation.
751  estimate = base::estimate_final_series_size(sparse_functor<>(&this->m_v1[0u],size1,&this->m_v2[0u],size2,retval));
752  // Correct the unlikely case of zero estimate.
753  if (unlikely(!estimate)) {
754  estimate = 1u;
755  }
756  // Get the number of threads to use.
757  // NOTE: tuning parameter here.
758  const unsigned n_threads = thread_pool::use_threads(
759  integer(size1) * size2,integer(500000L)
760  );
761  // Rehash the retun value's container accordingly. Check the tuning flag to see if we want to use
762  // multiple threads for initing the return value.
763  // NOTE: it is important here that we use the same n_threads for multiplication and memset as
764  // we tie together pinned threads with potentially different NUMA regions.
765  const unsigned n_threads_rehash = tuning::get_parallel_memory_set() ? n_threads : 1u;
766  // NOTE: if something goes wrong here, no big deal as retval is still empty.
767  retval.m_container.rehash(boost::numeric_cast<typename Series::size_type>(std::ceil(static_cast<double>(estimate) /
768  retval.m_container.max_load_factor())),n_threads_rehash);
769  piranha_assert(retval.m_container.bucket_count());
770  // NOTE: tuning parameter.
771  if ((integer(size1) * integer(size2)) / estimate > 200) {
772  dense_multiplication(retval,n_threads);
773  } else {
774  sparse_multiplication<sparse_functor<>>(retval,n_threads);
775  }
776  // Trace the result of estimation.
777  this->trace_estimates(retval.size(),estimate);
778  return retval;
779  }
780  // Utility function to determine block sizes.
781  static std::pair<integer,integer> get_block_sizes(const index_type &size1, const index_type &size2)
782  {
783  const integer block_size(512u), job_size = block_size.pow(2u);
784  // Rescale the block sizes according to the relative sizes of the input series.
785  auto block_size1 = (block_size * size1) / size2,
786  block_size2 = (block_size * size2) / size1;
787  // Avoid having zero block sizes, or block sizes exceeding job_size.
788  if (!block_size1) {
789  block_size1 = 1u;
790  } else if (block_size1 > job_size) {
791  block_size1 = job_size;
792  }
793  if (!block_size2) {
794  block_size2 = 1u;
795  } else if (block_size2 > job_size) {
796  block_size2 = job_size;
797  }
798  return std::make_pair(std::move(block_size1),std::move(block_size2));
799  }

842  // Dense multiplication method.
843  void dense_multiplication(Series &retval, const unsigned &n_threads) const
844  {
845  // Vectors of minimum / maximum values, cast to hardware int.
846  std::vector<value_type> mins;
847  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(mins),[](const std::pair<integer,integer> &p) {
848  return static_cast<value_type>(p.first);
849  });
850  std::vector<value_type> maxs;
851  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(maxs),[](const std::pair<integer,integer> &p) {
852  return static_cast<value_type>(p.second);
853  });
854  // Build the encoding vector.
855  std::vector<value_type> c_vec;
856  integer f_delta(1);
857  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(c_vec),
858  [&f_delta](const std::pair<integer,integer> &p) -> value_type {
859  auto old(f_delta);
860  f_delta *= p.second - p.first + 1;
861  return static_cast<value_type>(old);
862  });
863  // Try casting final delta.
864  (void)static_cast<value_type>(f_delta);
865  // Compute hmax and hmin.
866  piranha_assert(m_minmax_values.size() == c_vec.size());
867  const auto h_minmax = std::inner_product(m_minmax_values.begin(),m_minmax_values.end(),c_vec.begin(),
868  std::make_pair(integer(0),integer(0)),
869  [](const std::pair<integer,integer> &p1, const std::pair<integer,integer> &p2) {
870  return std::make_pair(p1.first + p2.first,p1.second + p2.second);
871  },[](const std::pair<integer,integer> &p, const value_type &value) {
872  return std::make_pair(p.first * value,p.second * value);
873  }
874  );
875  piranha_assert(f_delta == h_minmax.second - h_minmax.first + 1);
876  // Try casting hmax and hmin.
877  const auto hmin = static_cast<value_type>(h_minmax.first);
878  const auto hmax = static_cast<value_type>(h_minmax.second);
879  // Encoding functor.
880  typedef typename term_type::key_type::v_type unpack_type;
881  auto encoder = [&c_vec,hmin,&mins](const unpack_type &v) -> value_type {
882  piranha_assert(c_vec.size() == v.size());
883  piranha_assert(c_vec.size() == mins.size());
884  decltype(mins.size()) i = 0u;
885  value_type retval = std::inner_product(c_vec.begin(),c_vec.end(),v.begin(),value_type(0),
886  std::plus<value_type>(),[&i,&mins](const value_type &c, const value_type &n) -> value_type {
887  const decltype(mins.size()) old_i = i;
888  ++i;
889  piranha_assert(n >= mins[old_i]);
890  return static_cast<value_type>(c * (n - mins[old_i]));
891  }
892  );
893  return static_cast<value_type>(retval + hmin);
894  };
895  // Build copies of the input keys repacked according to the new Kronecker substitution. Attach
896  // also a pointer to the term.
897  typedef std::pair<value_type,term_type const *> new_key_type;
898  std::vector<new_key_type> new_keys1, new_keys2;
899  std::transform(this->m_v1.begin(),this->m_v1.end(),std::back_inserter(new_keys1),
900  [this,encoder](term_type const *ptr) {
901  return std::make_pair(encoder(ptr->m_key.unpack(this->m_s1->m_symbol_set)),ptr);
902  });
903  std::transform(this->m_v2.begin(),this->m_v2.end(),std::back_inserter(new_keys2),
904  [this,encoder](term_type const *ptr) {
905  return std::make_pair(encoder(ptr->m_key.unpack(this->m_s1->m_symbol_set)),ptr);
906  });
907  // Sort the the new keys.
908  std::stable_sort(new_keys1.begin(),new_keys1.end(),[](const new_key_type &p1, const new_key_type &p2) {
909  return p1.first < p2.first;
910  });
911  std::stable_sort(new_keys2.begin(),new_keys2.end(),[](const new_key_type &p1, const new_key_type &p2) {
912  return p1.first < p2.first;
913  });
914  // Store the sizes and compute the block sizes.
915  const index_type size1 = boost::numeric_cast<index_type>(new_keys1.size()),
916  size2 = boost::numeric_cast<index_type>(new_keys2.size());
917  piranha_assert(size1 == this->m_s1->size());
918  piranha_assert(size2 == this->m_s2->size());
919  const auto bsizes = get_block_sizes(size1,size2);
920  // Cast to hardware integers.
921  const auto bsize1 = static_cast<index_type>(bsizes.first), bsize2 = static_cast<index_type>(bsizes.second);
922  // Build the list of tasks.
923  dts_type<new_key_type> dense_task_sorter(new_keys1,new_keys2);
924  std::multiset<task_type,dts_type<new_key_type>> task_list(dense_task_sorter);
925  decltype(task_list.insert(std::declval<task_type>())) ins_result;
926  auto dense_task_from_indices = [hmin,hmax,&new_keys1,&new_keys2](const index_type &i_start, const index_type &i_end,
927  const index_type &j_start, const index_type &j_end) -> task_type
928  {
929  piranha_assert(i_end <= new_keys1.size() && j_end <= new_keys2.size());
930  piranha_assert(i_start < i_end && j_start < j_end);
931  const bucket_size_type a = boost::numeric_cast<bucket_size_type>((new_keys1[i_start].first + new_keys2[j_start].first) - hmin);
932  const bucket_size_type b = boost::numeric_cast<bucket_size_type>((new_keys1[i_end - 1u].first + new_keys2[j_end - 1u].first) - hmin);
933  piranha_assert(a <= b);
934  piranha_assert(b <= boost::numeric_cast<bucket_size_type>(hmax - hmin));
935  return task_type{std::make_pair(i_start,i_end),std::make_pair(j_start,j_end),
936  std::make_pair(a,b),std::make_pair(bucket_size_type(0u),bucket_size_type(0u)),false};
937  };
938  for (index_type i = 0u; i < size1 / bsize1; ++i) {
939  for (index_type j = 0u; j < size2 / bsize2; ++j) {
940  ins_result = task_list.insert(dense_task_from_indices(i * bsize1,(i + 1u) * bsize1,j * bsize2,(j + 1u) * bsize2));
941  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
942  }
943  if (size2 % bsize2) {
944  ins_result = task_list.insert(dense_task_from_indices(i * bsize1,(i + 1u) * bsize1,(size2 / bsize2) * bsize2,size2));
945  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
946  }
947  }
948  if (size1 % bsize1) {
949  for (index_type j = 0u; j < size2 / bsize2; ++j) {
950  ins_result = task_list.insert(dense_task_from_indices((size1 / bsize1) * bsize1,size1,j * bsize2,(j + 1u) * bsize2));
951  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
952  }
953  if (size2 % bsize2) {
954  ins_result = task_list.insert(dense_task_from_indices((size1 / bsize1) * bsize1,size1,(size2 / bsize2) * bsize2,size2));
955  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
956  }
957  }
958  // Prepare the storage for multiplication.
959  // NOTE: init everything explicitly to zero, as we make no assumption about the value of a default-cted
960  // coefficient.
961  using cf_vector_type = std::vector<typename term_type::cf_type>;
962  cf_vector_type cf_vector(boost::numeric_cast<typename cf_vector_type::size_type>((hmax - hmin) + 1),
963  typename term_type::cf_type(0));
964  if (n_threads == 1u) {
965  // Single-thread multiplication.
966  const auto it_f = task_list.end();
967  for (auto it = task_list.begin(); it != it_f; ++it) {
968  const index_type i_start = it->m_b1.first, j_start = it->m_b2.first,
969  i_end = it->m_b1.second, j_end = it->m_b2.second;
970  piranha_assert(i_end > i_start && j_end > j_start);
971  for (index_type i = i_start; i < i_end; ++i) {
972  for (index_type j = j_start; j < j_end; ++j) {
973  const auto idx = (new_keys1[i].first + new_keys2[j].first) - hmin;
974  piranha_assert(idx < boost::numeric_cast<value_type>(cf_vector.size()));
975  math::multiply_accumulate(cf_vector[static_cast<decltype(cf_vector.size())>(idx)],
976  new_keys1[i].second->m_cf,new_keys2[j].second->m_cf);
977  }
978  }
979  }
980  } else {
981  // Set of busy bucket regions, ordered by starting point.
982  // NOTE: we do not need a multiset, as regions that compare equal (i.e., same starting point) will not exist
983  // at the same time in the set by design.
984  region_comparer rc;
985  std::set<region_type,region_comparer> busy_regions(rc);
986  // Synchronization.
987  std::mutex m;
988  std::condition_variable cond;
989  // Thread function.
990  auto thread_function = [&cond,&m,&task_list,&busy_regions,&new_keys1,&new_keys2,hmin,&cf_vector,this] () {
991  task_type task;
992  while (true) {
993  {
994  // First, lock down everything.
995  std::unique_lock<std::mutex> lock(m);
996  if (task_list.empty()) {
997  break;
998  }
999  // Look for a suitable task.
1000  const auto it_f = task_list.end();
1001  auto it = task_list.begin();
1002  for (; it != it_f; ++it) {
1003  piranha_assert(!it->m_second_region);
1004  // Check the region.
1005  if (this->region_checker(it->m_r1,busy_regions)) {
1006  try {
1007  // The region is ok, insert it and break the cycle.
1008  auto tmp = busy_regions.insert(it->m_r1);
1009  (void)tmp;
1010  piranha_assert(tmp.second);
1011  piranha_assert(this->region_set_checker(busy_regions));
1012  } catch (...) {
1013  // NOTE: the idea here is that in case of errors
1014  // we want to restore the original situation
1015  // as if nothing happened.
1016  this->cleanup_regions(busy_regions,*it);
1017  throw;
1018  }
1019  break;
1020  }
1021  }
1022  // We might have identified a suitable task, check it is not end().
1023  if (it == it_f) {
1024  // The thread can't do anything, will have to wait until something happens
1025  // and then re-identify a good task.
1026  cond.wait(lock);
1027  continue;
1028  }
1029  // Now we have a good task, pop it from the task list.
1030  task = *it;
1031  task_list.erase(it);
1032  }
1033  try {
1034  // Perform the multiplication on the selected task.
1035  const index_type i_start = task.m_b1.first, j_start = task.m_b2.first,
1036  i_end = task.m_b1.second, j_end = task.m_b2.second;
1037  piranha_assert(i_end > i_start && j_end > j_start);
1038  for (index_type i = i_start; i < i_end; ++i) {
1039  for (index_type j = j_start; j < j_end; ++j) {
1040  const auto idx = (new_keys1[i].first + new_keys2[j].first) - hmin;
1041  piranha_assert(idx < boost::numeric_cast<value_type>(cf_vector.size()));
1042  math::multiply_accumulate(cf_vector[static_cast<decltype(cf_vector.size())>(idx)],
1043  new_keys1[i].second->m_cf,new_keys2[j].second->m_cf);
1044  }
1045  }
1046  } catch (...) {
1047  // Re-acquire the lock.
1048  std::lock_guard<std::mutex> lock(m);
1049  // Cleanup the regions.
1050  this->cleanup_regions(busy_regions,task);
1051  // Notify all waiting threads that a region was removed from the busy set.
1052  cond.notify_all();
1053  throw;
1054  }
1055  {
1056  // Re-acquire the lock.
1057  std::lock_guard<std::mutex> lock(m);
1058  // Take out the regions in which we just wrote from the set of busy regions.
1059  this->cleanup_regions(busy_regions,task);
1060  // Notify all waiting threads that a region was removed from the busy set.
1061  cond.notify_all();
1062  }
1063  }
1064  };
1065  // NOTE: one of the fundamental requirements here is that we wait for pending
1066  // tasks, in case of errors, before getting out of this function: thread_function
1067  // contains references to local variables, if we get out of here before the tasks are finished
1068  // memory corruption will occur.
1069  future_list<decltype(thread_pool::enqueue(0u,thread_function))> f_list;
1070  try {
1071  for (unsigned i = 0u; i < n_threads; ++i) {
1072  // NOTE: enqueue() will either happen or it won't, we only care
1073  // about memory allocation errors in push_back() here. In such case,
1074  // push_back() will wait on the temporary future from enqueue
1075  // before returning the exception.
1076  f_list.push_back(thread_pool::enqueue(i,thread_function));
1077  }
1078  // First let's wait for everything to finish.
1079  f_list.wait_all();
1080  // Then, let's handle the exceptions.
1081  f_list.get_all();
1082  } catch (...) {
1083  // Make sure any pending task is finished -> this is for
1084  // the case the exception was thrown in the future creation
1085  // loop. It is safe to call this again in case the exception
1086  // being handled is generated by get_all(), as wait_all() will check
1087  // the validity of the future before calling wait().
1088  f_list.wait_all();
1089  throw;
1090  }
1091  }
1092  // Build the return value.
1093  // Append the final delta to the coding vector for use in the decoding routine.
1094  c_vec.push_back(static_cast<value_type>(f_delta));
1095  // Temp vector for decoding.
1096  std::vector<value_type> tmp_v;
1097  tmp_v.resize(boost::numeric_cast<decltype(tmp_v.size())>(this->m_s1->m_symbol_set.size()));
1098  piranha_assert(c_vec.size() - 1u == tmp_v.size());
1099  piranha_assert(mins.size() == tmp_v.size());
1100  auto decoder = [&tmp_v,&c_vec,&mins,&maxs](const value_type &n) {
1101  decltype(c_vec.size()) i = 0u;
1102  std::generate(tmp_v.begin(),tmp_v.end(),
1103  [&n,&i,&mins,&maxs,&c_vec]() -> value_type
1104  {
1105  auto retval = (n % c_vec[i + 1u]) / c_vec[i] + mins[i];
1106  piranha_assert(retval >= mins[i] && retval <= maxs[i]);
1107  ++i;
1108  return static_cast<value_type>(retval);
1109  });
1110  };
1111  const auto cf_size = cf_vector.size();
1112  term_type tmp_term;
1113  for (decltype(cf_vector.size()) i = 0u; i < cf_size; ++i) {
1114  if (!math::is_zero(cf_vector[i])) {
1115  tmp_term.m_cf = std::move(cf_vector[i]);
1116  decoder(boost::numeric_cast<value_type>(i));
1117  tmp_term.m_key = decltype(tmp_term.m_key)(tmp_v.begin(),tmp_v.end());
1118  retval.insert(std::move(tmp_term));
1119  }
1120  }
1121  }
1122  // Struct for extracting bucket indices.
1123  // NOTE: use this instead of a lambda, since boost transform iterator needs the function
1124  // object to be assignable.
1125  struct sparse_bi_extractor
1126  {
1127  // NOTE: in some setups Boost is apparently unable to deduce the result type
1128  // of the functor and needs this typedef in the transform iterator.
1129  using result_type = bucket_size_type;
1130  explicit sparse_bi_extractor(const Series *retval) : m_retval(retval) {}
1131  template <typename Term>
1132  result_type operator()(const Term *t) const
1133  {
1134  return m_retval->m_container._bucket_from_hash(t->hash());
1135  }
1136  const Series *m_retval;
1137  };
1138  template <typename Functor>
1139  void sparse_multiplication(Series &retval,const unsigned &n_threads) const
1140  {
1141  // Type representing multiplication tasks.
1142  using task_type = std::tuple<term_type const *,term_type const **,term_type const **>;
1143  // A couple of handy shortcuts.

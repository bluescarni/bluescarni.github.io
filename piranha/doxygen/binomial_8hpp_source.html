60 // Generic binomial implementation using the falling factorial. U must be an integer
61 // type, T can be anything that supports basic arithmetics. k must be non-negative.
62 template <typename T, typename U>
63 inline T generic_binomial(const T &x, const U &k)
64 {
65  const U zero(0), one(1);
66  if (generic_binomial_check_k(k,zero)) {
67  piranha_throw(std::invalid_argument,"negative k value in binomial coefficient");
68  }
69  // Zero at bottom results always in 1.
70  if (k == zero) {
71  return T(1);
72  }
73  T tmp(x), retval = x / T(k);
74  --tmp;
75  for (auto i = static_cast<U>(k - one); i >= one; --i, --tmp) {
76  retval *= tmp;
77  retval /= T(i);
78  }
79  return retval;
80 }

82 // Compute gamma(a)/(gamma(b) * gamma(c)), assuming a, b and c are not negative ints.
83 template <typename T>
84 inline T compute_3_gamma(const T &a, const T &b, const T &c)
85 {
86  // Here we should never enter with negative ints.
87  piranha_assert(a >= T(0) || std::trunc(a) != a);
88  piranha_assert(b >= T(0) || std::trunc(b) != b);
89  piranha_assert(c >= T(0) || std::trunc(c) != c);
90  const T pi = boost::math::constants::pi<T>();
91  T tmp0(0), tmp1(1);
92  if (a < T(0)) {
93  tmp0 -= std::lgamma(T(1) - a);
94  tmp1 *= pi / std::sin(a * pi);
95  } else {
96  tmp0 += std::lgamma(a);
97  }
98  if (b < T(0)) {
99  tmp0 += std::lgamma(T(1) - b);
100  tmp1 *= std::sin(b * pi) / pi;
101  } else {
102  tmp0 -= std::lgamma(b);
103  }
104  if (c < T(0)) {
105  tmp0 += std::lgamma(T(1) - c);
106  tmp1 *= std::sin(c * pi) / pi;
107  } else {
108  tmp0 -= std::lgamma(c);
109  }
110  return std::exp(tmp0) * tmp1;
111 }

113 // Implementation of the generalised binomial coefficient for floating-point types.
114 template <typename T>
115 inline T fp_binomial(const T &x, const T &y)
116 {
117  static_assert(std::is_floating_point<T>::value,"Invalid type for fp_binomial.");
118  if (unlikely(!std::isfinite(x) || !std::isfinite(y))) {
119  piranha_throw(std::invalid_argument,"cannot compute binomial coefficient with non-finite floating-point argument(s)");
120  }
121  const bool neg_int_x = std::trunc(x + T(1)) == (x + T(1)) && (x + T(1)) <= T(0),
122  neg_int_y = std::trunc(y + T(1)) == (y + T(1)) && (y + T(1)) <= T(0),
123  neg_int_x_y = std::trunc(x - y + T(1)) == (x - y + T(1)) && (x - y + T(1)) <= T(0);
124  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
125  switch (mask) {
126  case 0u:
127  // Case 0 is the non-special one, use the default implementation.
128  return compute_3_gamma(x + T(1),y + T(1),x - y + T(1));
129  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
130  case 2u:
131  case 4u:
132  // These are finite numerators with infinite denominators.
133  return T(0.);
134  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
135  case 3u:
136  {
137  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
138  // formula to make them finite.
139  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
140  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
141  // here.
142  const auto phase = std::pow(T(-1),x + T(1)) / std::pow(T(-1),y + T(1));
143  return compute_3_gamma(-y,-x,x - y + T(1)) * phase;
144  }
145  case 5u:
146  {
147  const auto phase = std::pow(T(-1),x - y + T(1)) / std::pow(T(-1),x + T(1));
148  return compute_3_gamma(-(x - y),y + T(1),-x) * phase;
149  }
150  }
151  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
152  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
153  return T(0);
154 }

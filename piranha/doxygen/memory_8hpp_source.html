88  const std::size_t delta = (orig_a_size == a_size) ? alignment : orig_a_size - a_size;

100  }
101  // Record the delta.
102  *(static_cast<unsigned char *>(ptr) - 1) = static_cast<unsigned char>(delta);

127 inline void *aligned_palloc(const std::size_t &alignment, const std::size_t &size)
128 {
129  // Platform-independent part: special values for alignment and size.
130  if (unlikely(size == 0u)) {
131  return nullptr;
132  }
133  if (alignment == 0u) {
134  void *ptr = std::malloc(size);
135  if (unlikely(ptr == nullptr)) {
136  piranha_throw(std::bad_alloc,);
137  }
138  return ptr;
139  }
140 #if defined(PIRANHA_HAVE_POSIX_MEMALIGN)
141  void *ptr;
142  const int retval = ::posix_memalign(&ptr,alignment,size);
143  if (unlikely(retval != 0)) {
144  piranha_throw(std::bad_alloc,);
145  }
146  return ptr;
147 #elif defined(_WIN32)
148  void *ptr = ::_aligned_malloc(size,alignment);
149  if (unlikely(ptr == nullptr)) {
150  piranha_throw(std::bad_alloc,);
151  }
152  return ptr;
153 #else
154  // return detail::cpp_aligned_alloc(alignment,size);
155  piranha_throw(not_implemented_error,"memory alignment primitives are not available");
156 #endif
157 }

214 template <typename T>
215 inline bool alignment_check(const std::size_t &alignment)
216 {
217  // Platform-independent checks.
218  if (alignment == 0u) {
219  return true;
220  }
221  // Alignment must be power of 2.
222  if (unlikely(static_cast<bool>(alignment & (alignment - 1u)))) {
223  return false;
224  }
225  // Alignment must not be less than the natural alignment of T. We just need the '<' check
226  // as we already know that alignment is either a multiple of alignof(T) or a divisor.
227  if (unlikely(alignment < alignof(typename std::decay<T>::type))) {
228  return false;
229  }
230 #if defined(PIRANHA_HAVE_POSIX_MEMALIGN)
231  // Extra check for posix_memalign requirements.
232  if (unlikely(static_cast<bool>(alignment % sizeof(void *)))) {
233  return false;
234  }
235 #endif
236  return true;
237 }

261 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
262 inline void parallel_value_init(T *ptr, const std::size_t &size, const unsigned &n_threads)
263 {
264  using ranges_vector = std::vector<std::pair<T *,T *>>;
265  using rv_size_type = typename ranges_vector::size_type;
266  if (unlikely(ptr == nullptr)) {
267  piranha_assert(!size);
268  return;
269  }
270  // Initing functor.
271  auto init_function = [](T *start, T *end, const unsigned &thread_idx, ranges_vector *rv) {
272  auto orig_start = start;
273  try {
274  for (; start != end; ++start) {
275  ::new (static_cast<void *>(start)) T();
276  }
277  } catch (...) {
278  // Cleanup.
279  for (; orig_start != start; ++orig_start) {
280  orig_start->~T();
281  }
282  // Re-throw.
283  throw;
284  }
285  // If the init was successful and we are in multithreaded mode, record
286  // the range that was inited.
287  if (rv != nullptr) {
288  (*rv)[static_cast<rv_size_type>(thread_idx)].first = orig_start;
289  (*rv)[static_cast<rv_size_type>(thread_idx)].second = end;
290  }
291  };
292  if (n_threads <= 1) {
293  init_function(ptr,ptr + size,0u,nullptr);
294  } else {
295  // Init the ranges vector with (ptr,ptr) pairs, so they are empty ranges.
296  ranges_vector inited_ranges(static_cast<rv_size_type>(n_threads),std::make_pair(ptr,ptr));
297  if (unlikely(inited_ranges.size() != n_threads)) {
298  piranha_throw(std::bad_alloc,);
299  }
300  // Work per thread.
301  const auto wpt = static_cast<std::size_t>(size / n_threads);
302  future_list<decltype(thread_pool::enqueue(0u,init_function,ptr,ptr,0u,&inited_ranges))> f_list;
303  try {
304  for (auto i = 0u; i < n_threads; ++i) {
305  auto start = ptr + i * wpt, end = (i == n_threads - 1u) ? ptr + size : ptr + (i + 1u) * wpt;
306  f_list.push_back(thread_pool::enqueue(i,init_function,start,end,i,&inited_ranges));
307  }
308  f_list.wait_all();
309  f_list.get_all();
310  } catch (...) {
311  f_list.wait_all();
312  // Rollback the ranges that were inited.
313  for (const auto &p: inited_ranges) {
314  for (auto start = p.first; start != p.second; ++start) {
315  start->~T();
316  }
317  }
318  throw;
319  }
320  }
321 }

338 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
339 inline void parallel_destroy(T *ptr, const std::size_t &size, const unsigned &n_threads)
340 {
341  using ranges_vector = std::vector<std::pair<T *,T *>>;
342  using rv_size_type = typename ranges_vector::size_type;
343  // Nothing to be done for null pointers.
344  if (unlikely(ptr == nullptr)) {
345  piranha_assert(!size);
346  return;
347  }
348  // Nothing needs to be done for trivially destructible types:
349  // http://en.cppreference.com/w/cpp/language/lifetime
350  if (std::is_trivially_destructible<T>::value) {
351  return;
352  }
353  // Destroy functor.
354  auto destroy_function = [](T *start, T *end) {
355  for (; start != end; ++start) {
356  start->~T();
357  }
358  };
359  if (n_threads <= 1u) {
360  destroy_function(ptr,ptr + size);
361  } else {
362  // A vector of ranges representing elements yet to be destroyed in case something goes wrong
363  // in the multithreaded part.
364  ranges_vector d_ranges;
365  future_list<decltype(thread_pool::enqueue(0u,destroy_function,ptr,ptr))> f_list;
366  try {
367  d_ranges.resize(static_cast<rv_size_type>(n_threads),std::make_pair(ptr,ptr));
368  if (unlikely(d_ranges.size() != n_threads)) {
369  piranha_throw(std::bad_alloc,);
370  }
371  } catch (...) {
372  // Just perform the single-threaded version, and GTFO.
373  destroy_function(ptr,ptr + size);
374  return;
375  }
376  try {
377  // Work per thread.
378  const auto wpt = static_cast<std::size_t>(size / n_threads);
379  // The ranges to destroy in the cleanup phase are, initially, all the ranges.
380  for (auto i = 0u; i < n_threads; ++i) {
381  auto start = ptr + i * wpt, end = (i == n_threads - 1u) ? ptr + size : ptr + (i + 1u) * wpt;
382  d_ranges[static_cast<rv_size_type>(i)] = std::make_pair(start,end);
383  }
384  // Perform the actual destruction and update the d_ranges vector.
385  for (auto i = 0u; i < n_threads; ++i) {
386  auto start = d_ranges[static_cast<rv_size_type>(i)].first, end = d_ranges[static_cast<rv_size_type>(i)].second;
387  f_list.push_back(thread_pool::enqueue(i,destroy_function,start,end));
388  // The range needs not to be destroyed anymore, as the enqueue/push_back was successful. Replace
389  // with an empty range.
390  d_ranges[static_cast<rv_size_type>(i)].first = ptr;
391  d_ranges[static_cast<rv_size_type>(i)].second = ptr;
392  }
393  f_list.wait_all();
394  // NOTE: T is a container_element, no need to get exceptions here.
395  } catch (...) {
396  f_list.wait_all();
397  // If anything failed in the multithreaded part, just destroy in single-thread the ranges
398  // that were not destroyed.
399  for (const auto &p: d_ranges) {
400  destroy_function(p.first,p.second);
401  }
402  }
403  }
404 }

410 template <typename T>
411 class parallel_deleter

413  public:

455 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
456 inline std::unique_ptr<T[],detail::parallel_deleter<T>> make_parallel_array(const std::size_t &size, const unsigned &n_threads)
457 {
458  if (unlikely(size > std::numeric_limits<std::size_t>::max() / sizeof(T))) {
459  piranha_throw(std::bad_alloc,);
460  }
461  // Allocate space. This could be nullptr if size is zero.
462  auto ptr = static_cast<T *>(aligned_palloc(0u,static_cast<std::size_t>(size * sizeof(T))));
463  try {
464  // No problems here with nullptr, will be a no-op.
465  parallel_value_init(ptr,size,n_threads);
466  } catch (...) {
467  piranha_assert(ptr != nullptr);
468  // Free the allocated memory. This is noexcept.
469  aligned_pfree(0u,static_cast<void *>(ptr));
470  throw;
471  }
472  return std::unique_ptr<T[],detail::parallel_deleter<T>>(ptr,detail::parallel_deleter<T>{size,n_threads});
473 }

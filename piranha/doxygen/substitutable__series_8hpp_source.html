80  template <typename Term, typename T>
81  struct subs_term_score
82  {
83  static const unsigned value = static_cast<unsigned>(has_subs<typename Term::cf_type,T>::value) +
84  (static_cast<unsigned>(key_has_subs<typename Term::key_type,T>::value) << 1u);

86  // Case 1: subs only on cf.
87  template <typename T, typename Term>
88  using cf_subs_type = decltype(math::subs(std::declval<typename Term::cf_type const &>(),std::declval<std::string const &>(),
89  std::declval<T const &>()));
90  template <typename T, typename Term>
91  using ret_type_1 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<Derived const &>());
92  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 1u,int>::type = 0>
93  static ret_type_1<T,Term> subs_term_impl(const Term &t, const std::string &name, const T &x, const symbol_set &s_set)
94  {
95  Derived tmp;
96  tmp.set_symbol_set(s_set);
97  tmp.insert(Term(typename Term::cf_type(1),t.m_key));
98  // NOTE: use moves here in case the multiplication can take advantage.
99  return math::subs(t.m_cf,name,x) * std::move(tmp);
100  }
101  // Case 2: subs only on key.
102  template <typename T, typename Term>
103  using k_subs_type = typename decltype(std::declval<const typename Term::key_type &>().subs(std::declval<const std::string &>(),
104  std::declval<const T &>(),std::declval<const symbol_set &>()))::value_type::first_type;
105  template <typename T, typename Term>
106  using ret_type_2_ = decltype(std::declval<Derived const &>() * std::declval<const k_subs_type<T,Term> &>());
107  template <typename T, typename Term>
108  using ret_type_2 = typename std::enable_if<is_addable_in_place<ret_type_2_<T,Term>>::value &&
109  std::is_constructible<ret_type_2_<T,Term>,int>::value,ret_type_2_<T,Term>>::type;
110  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 2u,int>::type = 0>
111  static ret_type_2<T,Term> subs_term_impl(const Term &t, const std::string &name, const T &x, const symbol_set &s_set)
112  {
113  ret_type_2<T,Term> retval(0);
114  auto ksubs = t.m_key.subs(name,x,s_set);
115  for (auto &p: ksubs) {
116  Derived tmp;
117  tmp.set_symbol_set(s_set);
118  tmp.insert(Term{t.m_cf,std::move(p.second)});
119  // NOTE: possible use of multadd here in the future.
120  retval += std::move(tmp) * std::move(p.first);
121  }
122  return retval;
123  }
124  // Case 3: subs on cf and key.
125  // NOTE: the checks on type 2 are already present in the alias above.
126  template <typename T, typename Term>
127  using ret_type_3 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<const ret_type_2<T,Term> &>());
128  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 3u,int>::type = 0>
129  static ret_type_3<T,Term> subs_term_impl(const Term &t, const std::string &name, const T &x, const symbol_set &s_set)
130  {
131  // Accumulator for the sum below. This is the same type resulting from case 2.
132  ret_type_2<T,Term> acc(0);
133  auto ksubs = t.m_key.subs(name,x,s_set);
134  auto cf_subs = math::subs(t.m_cf,name,x);
135  for (auto &p: ksubs) {
136  Derived tmp;
137  tmp.set_symbol_set(s_set);
138  tmp.insert(Term(typename Term::cf_type(1),std::move(p.second)));
139  // NOTE: multadd chance.
140  acc += std::move(tmp) * std::move(p.first);
141  }
142  return std::move(cf_subs) * std::move(acc);
143  }

252  template <typename T>
253  typename std::enable_if<detail::is_mp_rational<T>::value,T>::type convert_to_impl() const
254  {
255  if (is_nan()) {
256  piranha_throw(std::overflow_error,"cannot convert NaN to rational");
257  }
258  if (is_inf()) {
259  piranha_throw(std::overflow_error,"cannot convert infinity to rational");
260  }
261  if (sign() == 0) {
262  return T{};
263  }
264  // Get string representation.
265  ::mpfr_exp_t exp(0);
266  char *cptr = ::mpfr_get_str(nullptr,&exp,10,0,m_value,default_rnd);
267  if (unlikely(!cptr)) {
268  piranha_throw(std::overflow_error,"error in conversion of real to rational: the call to the MPFR function failed");
269  }
270  smart_mpfr_str str_ptr(cptr,::mpfr_free_str);
271  // Transform into fraction.
272  std::size_t digits = 0u;
273  for (; *cptr != '\0'; ++cptr) {
274  if (detail::is_digit(*cptr)) {
275  ++digits;
276  }
277  }
278  if (!digits) {
279  piranha_throw(std::overflow_error,"error in conversion of real to rational: invalid number of digits");
280  }
281  // NOTE: here the only exception that can be thrown is when raising to a power
282  // that cannot be represented by unsigned long.
283  try {
284  T retval(str_ptr.get());
285  // NOTE: possible optimizations here include going through direct GMP routines.
286  retval *= T(1,10).pow(digits);
287  retval *= T(10).pow(exp);
288  return retval;
289  } catch (...) {
290  piranha_throw(std::overflow_error,"error in conversion of real to rational: exponent is too large");
291  }
292  }

1940 // Compute gamma(a)/(gamma(b) * gamma(c)), assuming a, b and c are not negative ints. The logarithm
1941 // of the gamma function is used internally.
1942 inline real real_compute_3_gamma(const real &a, const real &b, const real &c, const ::mpfr_prec_t &prec)
1943 {
1944  // Here we should never enter with negative ints.
1945  piranha_assert(a.sign() >= 0 || a.truncated() != a);
1946  piranha_assert(b.sign() >= 0 || b.truncated() != b);
1947  piranha_assert(c.sign() >= 0 || c.truncated() != c);
1948  const real pi = real{0,prec}.pi();
1949  real tmp0(0), tmp1(1);
1950  if (a.sign() < 0) {
1951  tmp0 -= (1 - a).lgamma();
1952  tmp1 *= pi / (a * pi).sin();
1953  } else {
1954  tmp0 += a.lgamma();
1955  }
1956  if (b.sign() < 0) {
1957  tmp0 += (1 - b).lgamma();
1958  tmp1 *= (b * pi).sin() / pi;
1959  } else {
1960  tmp0 -= b.lgamma();
1961  }
1962  if (c.sign() < 0) {
1963  tmp0 += (1 - c).lgamma();
1964  tmp1 *= (c * pi).sin() / pi;
1965  } else {
1966  tmp0 -= c.lgamma();
1967  }
1968  return tmp0.exp() * tmp1;
1969 }

1988 inline real real::binomial(const real &y) const
1989 {
1990  if (unlikely(is_nan() || is_inf() || y.is_nan() || y.is_inf())) {
1991  piranha_throw(std::invalid_argument,"cannot compute binomial coefficient with non-finite real argument(s)");
1992  }
1993  // Work with the max precision.
1994  const ::mpfr_prec_t max_prec = std::max< ::mpfr_prec_t>(get_prec(),y.get_prec());
1995  const bool neg_int_x = truncated() == (*this) && sign() < 0,
1996  neg_int_y = y.truncated() == y && y.sign() < 0,
1997  neg_int_x_y = ((*this) - y).truncated() == ((*this) - y) && ((*this) - y).sign() < 0;
1998  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
1999  switch (mask) {
2000  case 0u:
2001  // Case 0 is the non-special one, use the default implementation.
2002  return detail::real_compute_3_gamma((*this) + 1,y + 1,(*this) - y + 1,max_prec);
2003  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
2004  case 2u:
2005  case 4u:
2006  // These are finite numerators with infinite denominators.
2007  return real{0,max_prec};
2008  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
2009  case 3u:
2010  {
2011  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
2012  // formula to make them finite.
2013  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
2014  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
2015  // here.
2016  const auto phase = math::pow(-1,(*this) + 1) / math::pow(-1,y + 1);
2017  return detail::real_compute_3_gamma(-y,-(*this),(*this) - y + 1,max_prec) * phase;
2018  }
2019  case 5u:
2020  {
2021  const auto phase = math::pow(-1,(*this) - y + 1) / math::pow(-1,(*this) + 1);
2022  return detail::real_compute_3_gamma(-((*this) - y),y + 1,-(*this),max_prec) * phase;
2023  }
2024  }
2025  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
2026  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
2027  return real{0,max_prec};
2028 }

84  bool check() const
85  {
86  // Check for sorted range.
87  if (!std::is_sorted(begin(),end())) {
88  return false;
89  }
90  if (size() < 2u) {
91  return true;
92  }
93  // Check for duplicates.
94  for (size_type i = 0u; i < size() - 1u; ++i) {
95  if (m_values[i] == m_values[i + 1u]) {
96  return false;
97  }
98  }
99  return true;
100  }

102  template <typename Iterator, typename Symbol>
103  using it_ctor_enabler = typename std::enable_if<is_input_iterator<Iterator>::value &&
104  std::is_constructible<Symbol,decltype(*(std::declval<const Iterator &>()))>::value,int>::type;
105  // Serialization support.
106  friend class boost::serialization::access;
107  template <class Archive>
108  void save(Archive &ar, unsigned int) const
109  {
110  const auto size = m_values.size();
111  ar & size;
112  for (const auto &s: m_values) {
113  // Store the names.
114  ar & s.get_name();

117  template <class Archive>
118  void load(Archive &ar, unsigned int)
119  {
120  // A new, empty set.
121  symbol_set ss;
122  // First recover the size.
123  std::vector<symbol>::size_type size;
124  ar & size;
125  // Restore using strings.
126  // NOTE: this is the slowest and safest way of doing it, even in face
127  // of bogus input from the archive (e.g., out-of-order symbols, duplicate names,
128  // etc.). This can certainly be optimised in the future.
129  std::string name;
130  for (decltype(size) i = 0u; i < size; ++i) {
131  ar & name;
132  ss.add(symbol(name));
133  }
134  // Move in the new set.
135  *this = std::move(ss);
136  }

417  void add(const symbol &s)
418  {
419  // Copy it to provide strong exception safety.
420  std::vector<symbol> new_values;
421  new_values.reserve(size() + size_type(1u));
422  std::copy(begin(),end(),std::back_inserter(new_values));
423  const auto it = std::lower_bound(new_values.begin(),new_values.end(),s);
424  if (unlikely(it != new_values.end() && *it == s)) {
425  piranha_throw(std::invalid_argument,"symbol already present in this set");
426  }
427  new_values.insert(it,s);
428  // Move in the new args vector.
429  m_values = std::move(new_values);
430  piranha_assert(check());
431  }

551  size_type index_of(const symbol &s) const
552  {
553  // Need to check for potential overflows here.
554  using diff_type = std::iterator_traits<const_iterator>::difference_type;
555  using udiff_type = std::make_unsigned<diff_type>::type;
556  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
557  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
558  piranha_throw(std::overflow_error,"potential overflow in the computaion of the "
559  "index of a symbol");
560  }
561  const auto it = std::lower_bound(m_values.begin(),m_values.end(),s);
562  if (it != m_values.end() && *it == s) {
563  auto retval = it - m_values.begin();
564  piranha_assert(retval >= diff_type(0));
565  return static_cast<size_type>(retval);
566  }
567  return m_values.size();
568  }

595 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
596 {
597  size_type ia = 0u, ib = 0u;
598  while (true) {
599  if (ia == a.size() || ib == b.size()) {
600  break;
601  }
602  if (a[ia] == b[ib]) {
603  m_values.push_back(ia);
604  ++ia;
605  ++ib;
606  } else if (a[ia] < b[ib]) {
607  ++ia;
608  } else {
609  ++ib;
610  }
611  }
612 }

625 template <typename T>
626 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol,T> &map)
627 {
628  for (const auto &p: map) {
629  const auto idx = a.index_of(p.first);
630  if (idx != a.size()) {
631  m_pairs.push_back(std::make_pair(idx,p.second));
632  }
633  }
634  std::sort(m_pairs.begin(),m_pairs.end(),[](const value_type &p1, const value_type &p2) {
635  return p1.first < p2.first;
636  });
637  // Check that there are no duplicate positions.
638  piranha_assert(std::unique(m_pairs.begin(),m_pairs.end(),[](const value_type &p1, const value_type &p2) {

80  typedef SignedInteger int_type;
81  private:
82 #if !defined(PIRANHA_DOXYGEN_INVOKED)
83  static_assert(detail::ka_type_reqs<int_type>::value,"This class can be used only with signed integers.");
84  // This is a 4-tuple of int_type built as follows:
85  // 0. vector of absolute values of the upper/lower limit for each component,
86  // 1. h_min,
87  // 2. h_max,
88  // 3. h_max - h_min.

90  // Vector of limits.
91  typedef std::vector<limit_type> limits_type;

99  typedef std::size_t size_type;
100  private:
101  // Static vector of limits built at startup.
102  // NOTE: here we should not have problems when interoperating with libraries that modify the GMP allocation functions,
103  // as we do not store any static piranha::integer: the creation and destruction of integer objects is confined to the determine_limit()
104  // function.
105  static const limits_type m_limits;
106  // Determine limits for m-dimensional vectors.
107  // NOTE: when reasoning about this, keep in mind that this is not a completely generic
108  // codification: min/max vectors are negative/positive and symmetric. This makes it easy
109  // to reason about overflows during (de)codification of vectors, representability of the
110  // quantities involved, etc.
111  static limit_type determine_limit(const size_type &m)
112  {
113  piranha_assert(m >= 1u);
114  std::mt19937 engine(static_cast<unsigned long>(m));
115  std::uniform_int_distribution<int> dist(-5,5);
116  // Perturb integer value: add random quantity and then take next prime.
117  auto perturb = [&engine,&dist] (integer &arg) {
118  arg += (dist(engine) * (arg)) / 100;
119  arg = arg.nextprime();
120  };
121  // Build initial minmax and coding vectors: all elements in the [-1,1] range.
122  std::vector<integer> m_vec, M_vec, c_vec, prev_c_vec, prev_m_vec, prev_M_vec;
123  c_vec.push_back(integer(1));
124  m_vec.push_back(integer(-1));
125  M_vec.push_back(integer(1));
126  for (size_type i = 1u; i < m; ++i) {
127  m_vec.push_back(integer(-1));
128  M_vec.push_back(integer(1));
129  c_vec.push_back(c_vec.back() * integer(3));
130  }
131  // Functor for scalar product of two vectors.
132  auto dot_prod = [](const std::vector<integer> &v1, const std::vector<integer> &v2) -> integer {
133  piranha_assert(v1.size() && v1.size() == v2.size());
134  return std::inner_product(v1.begin(),v1.end(),v2.begin(),integer(0));
135  };
136  while (true) {
137  // Compute the current h_min/max and diff.
138  integer h_min = dot_prod(c_vec,m_vec);
139  integer h_max = dot_prod(c_vec,M_vec);
140  integer diff = h_max - h_min;
141  piranha_assert(diff >= 0);
142  try {
143  // Try to cast everything to hardware integers.
144  (void)static_cast<int_type>(h_min);
145  (void)static_cast<int_type>(h_max);
146  // Here it is +1 because h_max - h_min must be strictly less than the maximum value
147  // of int_type. In the paper, in eq. (7), the Delta_i product appearing in the
148  // decoding of the last component of a vector is equal to (h_max - h_min + 1) so we need
149  // to be able to represent it.
150  (void)static_cast<int_type>(diff + 1);
151  // NOTE: we do not need to cast the individual elements of m/M vecs, as the representability
152  // of h_min/max ensures the representability of m/M as well.
153  } catch (const std::overflow_error &) {
154  std::vector<int_type> tmp;
155  // Check if we are at the first iteration.
156  if (prev_c_vec.size()) {
157  const auto h_min = dot_prod(prev_c_vec,prev_m_vec), h_max = dot_prod(prev_c_vec,prev_M_vec);
158  std::transform(prev_M_vec.begin(),prev_M_vec.end(),std::back_inserter(tmp),[](const integer &n) {
159  return static_cast<int_type>(n);
160  });
161  return std::make_tuple(
162  tmp,
163  static_cast<int_type>(h_min),
164  static_cast<int_type>(h_max),
165  static_cast<int_type>(h_max - h_min)
166  );
167  } else {
168  // Here it means m variables are too many, and we stopped at the first iteration
169  // of the cycle. Return tuple filled with zeroes.
170  return std::make_tuple(tmp,int_type(0),int_type(0),int_type(0));
171  }
172  }
173  // Store old vectors.
174  prev_c_vec = c_vec;
175  prev_m_vec = m_vec;
176  prev_M_vec = M_vec;
177  // Generate new coding vector for next iteration.
178  auto it = c_vec.begin() + 1, prev_it = prev_c_vec.begin();
179  for (; it != c_vec.end(); ++it, ++prev_it) {
180  // Recover original delta.
181  *it /= *prev_it;
182  // Multiply by two and perturb.
183  *it *= 2;
184  perturb(*it);
185  // Multiply by the new accumulated delta product.
186  *it *= *(it - 1);
187  }
188  // Fill in the minmax vectors, apart from the last component.
189  it = c_vec.begin() + 1;
190  piranha_assert(M_vec.size() && M_vec.size() == m_vec.size());
191  for (size_type i = 0u; i < M_vec.size() - 1u; ++i, ++it) {
192  M_vec[i] = ((*it) / *(it - 1) - 1) / 2;
193  m_vec[i] = -M_vec[i];
194  }
195  // We need to generate the last interval, which does not appear in the coding vector.
196  // Take the previous interval and enlarge it so that the corresponding delta is increased by a
197  // perturbed factor of 2.
198  M_vec.back() = (4 * M_vec.back() + 1) / 2;
199  perturb(M_vec.back());
200  m_vec.back() = -M_vec.back();
201  }
202  }

204  {
205  limits_type retval;
206  retval.push_back(std::make_tuple(std::vector<int_type>{},int_type(0),int_type(0),int_type(0)));
207  for (size_type i = 1u; ; ++i) {
208  const auto tmp = determine_limit(i);
209  if (std::get<0u>(tmp).empty()) {
210  break;
211  } else {
212  retval.push_back(tmp);
213  }

261  template <typename Vector>
262  static int_type encode(const Vector &v)
263  {
264  const auto size = v.size();
265  // NOTE: here the check is >= because indices in the limits vector correspond to the sizes of the vectors to be encoded.
266  if (unlikely(size >= m_limits.size())) {
267  piranha_throw(std::invalid_argument,"size of vector to be encoded is too large");
268  }
269  if (unlikely(!size)) {
270  return int_type(0);
271  }
272  // Cache quantities.
273  const auto &limit = m_limits[size];
274  const auto &minmax_vec = std::get<0u>(limit);
275  // Check that the vector's components are compatible with the limits.
276  // NOTE: here size is not greater than m_limits.size(), which in turn is compatible with the minmax vectors.
277  for (min_int<decltype(v.size()),decltype(minmax_vec.size())> i = 0u; i < size; ++i) {
278  if (unlikely(boost::numeric_cast<int_type>(v[i]) < -minmax_vec[i] || boost::numeric_cast<int_type>(v[i]) > minmax_vec[i])) {
279  piranha_throw(std::invalid_argument,"a component of the vector to be encoded is out of bounds");
280  }
281  }
282  piranha_assert(minmax_vec[0u] > 0);
283  int_type retval = static_cast<int_type>(boost::numeric_cast<int_type>(v[0u]) + minmax_vec[0u]),
284  cur_c = static_cast<int_type>(2 * minmax_vec[0u] + 1);
285  piranha_assert(retval >= 0);
286  for (decltype(v.size()) i = 1u; i < size; ++i) {
287  retval = static_cast<int_type>(retval + ((boost::numeric_cast<int_type>(v[i]) + minmax_vec[i]) * cur_c));
288  piranha_assert(minmax_vec[i] > 0);
289  cur_c = static_cast<int_type>(cur_c * (2 * minmax_vec[i] + 1));
290  }
291  return static_cast<int_type>(retval + std::get<1u>(limit));
292  }

315  template <typename Vector>
316  static void decode(Vector &retval, const int_type &n)
317  {
318  typedef typename Vector::value_type v_type;
319  const auto m = retval.size();
320  if (unlikely(m >= m_limits.size())) {
321  piranha_throw(std::invalid_argument,"size of vector to be decoded is too large");
322  }
323  if (unlikely(!m)) {
324  if (unlikely(n != 0)) {
325  piranha_throw(std::invalid_argument,"a vector of size 0 must always be encoded as 0");
326  }
327  return;
328  }
329  // Cache values.
330  const auto &limit = m_limits[m];
331  const auto &minmax_vec = std::get<0u>(limit);
332  const auto hmin = std::get<1u>(limit), hmax = std::get<2u>(limit);
333  if (unlikely(n < hmin || n > hmax)) {
334  piranha_throw(std::invalid_argument,"the integer to be decoded is out of bounds");
335  }
336  // NOTE: the static_cast here is useful when working with int_type == char. In that case,
337  // the binary operation on the RHS produces an int (due to integer promotion rules), which gets
338  // assigned back to char causing the compiler to complain about potentially lossy conversion.
339  const int_type code = static_cast<int_type>(n - hmin);
340  piranha_assert(code >= 0);
341  piranha_assert(minmax_vec[0u] > 0);
342  int_type mod_arg = static_cast<int_type>(2 * minmax_vec[0u] + 1);
343  // Do the first value manually.
344  retval[0u] = boost::numeric_cast<v_type>((code % mod_arg) - minmax_vec[0u]);
345  for (min_int<typename Vector::size_type,decltype(minmax_vec.size())> i = 1u; i < m; ++i) {
346  piranha_assert(minmax_vec[i] > 0);
347  retval[i] = boost::numeric_cast<v_type>((code % (mod_arg * (2 * minmax_vec[i] + 1))) / mod_arg - minmax_vec[i]);
348  mod_arg = static_cast<int_type>(mod_arg * (2 * minmax_vec[i] + 1));
349  }
350  }

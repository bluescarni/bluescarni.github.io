483 {
484  // A couple of handy aliases.
485  // NOTE: we need both bso_type and common_type because bso gives access to the ::value
486  // member to decide on the implementation, common_type is handy because it's shorter
487  // than using ::type on the bso.
488  template <typename T, typename U, int N>
489  using bso_type = detail::binary_series_op_return_type<typename std::decay<T>::type,
490  typename std::decay<U>::type,N>;
491  template <typename T, typename U, int N>
492  using series_common_type = typename bso_type<T,U,N>::type;
493  // Main implementation function for add/sub: all the possible cases end up here eventually, after
494  // any necessary conversion.
495  template <bool Sign, typename T, typename U>
496  static series_common_type<T,U,0> binary_add_impl(T &&x, U &&y)
497  {
498  // This is the same as T and U.
499  using ret_type = series_common_type<T,U,0>;
500  static_assert(std::is_same<typename std::decay<T>::type,ret_type>::value,"Invalid type.");
501  static_assert(std::is_same<typename std::decay<U>::type,ret_type>::value,"Invalid type.");
502  // Init the return value from the first operand.
503  // NOTE: this is always possible to do, a series is always copy/move-constructible.
504  ret_type retval(std::forward<T>(x));
505  // NOTE: here we cover the corner case in which x and y are the same object.
506  // This could be problematic in the algorithm below if retval was move-cted from
507  // x, since in such a case y will also be erased. This will happen if one does
508  // std::move(a) + std::move(a)
509  if (unlikely(&x == &y)) {
510  retval.template merge_terms<Sign>(retval);
511  return retval;
512  }
513  // NOTE: we do not use x any more, as it might have been moved.
514  if (likely(retval.m_symbol_set == y.m_symbol_set)) {
515  retval.template merge_terms<Sign>(std::forward<U>(y));
516  } else {
517  // Let's fix the args of the first series, if needed.
518  auto merge = retval.m_symbol_set.merge(y.m_symbol_set);
519  if (merge != retval.m_symbol_set) {
520  // This is a move assignment, always possible.
521  retval = retval.merge_arguments(merge);
522  }
523  // Fix the args of the second series.
524  if (merge != y.m_symbol_set) {
525  retval.template merge_terms<Sign>(y.merge_arguments(merge));
526  } else {
527  retval.template merge_terms<Sign>(std::forward<U>(y));
528  }
529  }
530  return retval;
531  }
532  // NOTE: this case has no special algorithmic requirements, the base requirements for cf and series types
533  // already cover everything (including the needeed series constructors and cf arithmetic operators).
534  template <typename T, typename U, typename std::enable_if<bso_type<T,U,0>::value == 0u,int>::type = 0>
535  static series_common_type<T,U,0> dispatch_binary_add(T &&x, U &&y)
536  {
537  return binary_add_impl<true>(std::forward<T>(x),std::forward<U>(y));
538  }
539  // NOTE: this covers two cases:
540  // - equal recursion, first series wins,
541  // - first higher recursion, coefficient of first series wins,
542  // In both cases we need to construct a T from y.
543  template <typename T, typename U, typename std::enable_if<(bso_type<T,U,0>::value == 1u || bso_type<T,U,0>::value == 4u) &&
544  // In order for the implementation to work, we need to be able to build T from U.
545  std::is_constructible<typename std::decay<T>::type,const typename std::decay<U>::type &>::value,
546  int>::type = 0>
547  static series_common_type<T,U,0> dispatch_binary_add(T &&x, U &&y)
548  {
549  typename std::decay<T>::type y1(std::forward<U>(y));
550  return dispatch_binary_add(std::forward<T>(x),std::move(y1));
551  }
552  // Symmetric of 1.
553  template <typename T, typename U, typename std::enable_if<bso_type<T,U,0>::value == 2u,int>::type = 0>
554  static auto dispatch_binary_add(T &&x, U &&y) -> decltype(dispatch_binary_add(std::forward<U>(y),std::forward<T>(x)))
555  {
556  return dispatch_binary_add(std::forward<U>(y),std::forward<T>(x));
557  }
558  // Two cases:
559  // - equal series recursion, 3rd coefficient type generated,
560  // - first higher recursion, coefficient result is different from first series.
561  // In both cases we need to promote both operands to a 3rd type.
562  template <typename T, typename U, typename std::enable_if<(bso_type<T,U,0>::value == 3u || bso_type<T,U,0>::value == 5u) &&
563  // In order for the implementation to work, we need to be able to build the common type from T and U.
564  std::is_constructible<series_common_type<T,U,0>,const typename std::decay<T>::type &>::value &&
565  std::is_constructible<series_common_type<T,U,0>,const typename std::decay<U>::type &>::value,
566  int>::type = 0>
567  static series_common_type<T,U,0> dispatch_binary_add(T &&x, U &&y)
568  {
569  series_common_type<T,U,0> x1(std::forward<T>(x));
570  series_common_type<T,U,0> y1(std::forward<U>(y));
571  return dispatch_binary_add(std::move(x1),std::move(y1));
572  }
573  // 6 and 7 are symmetric cases of 4 and 5. Just reverse the operands.
574  template <typename T, typename U, typename std::enable_if<bso_type<T,U,0>::value == 6u || bso_type<T,U,0>::value == 7u,int>::type = 0>
575  static auto dispatch_binary_add(T &&x, U &&y) -> decltype(dispatch_binary_add(std::forward<U>(y),std::forward<T>(x)))
576  {
577  return dispatch_binary_add(std::forward<U>(y),std::forward<T>(x));
578  }
579  // NOTE: here we use the version of binary_add with const references. The idea
580  // here is that any overload other than the const references one is an optimisation detail
581  // and that for the operator to be enabled the "canonical" form of addition operator must be available.
582  // The consequence is that if, for instance, only a move constructor is available, the operator

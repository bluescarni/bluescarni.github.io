48 // This is essentially a reduced std::vector replacement that should use less storage on most implementations
49 // (e.g., it has a size of 16 on Linux 64-bit and it _should_ have a size of 8 on many 32-bit archs).
50 // NOTE: earlier versions of this class used to have support for custom allocation. See, e.g., commit
51 // 6e1dd235c729ed05eb6e872dceece2f20a624f32
52 // It seems however that there are some defects currently in the standard C++ library that make it quite hard
53 // to implement reasonably strong exception safety and nothrow guarantees. See, e.g.,
54 // http://cplusplus.github.io/LWG/lwg-defects.html#2103
55 // It seems like in the current standard it is not possible to implemement noexcept move assignment not even
56 // for std::alloc. The crux of the matter is that if propagate_on_container_move_assignment is not defined,
57 // we will need in general a memory allocation (that can fail).
58 // Another, less important, problem is that for copy assignment the presence of
59 // propagate_on_container_copy_assignment seems to force one to nuke pre-emptively the content of the container
60 // before doing anything (thus preventing strong forms of exception safety).
61 // The approach taken here is thus to just hard-code std::allocator, which is stateless, and force nothrow
62 // behaviour when doing assignments. This should work with all stateless allocators (including our own allocator).
63 // Unfortunately it is not clear if the extension allocators from GCC are stateless, so better not to muck
64 // with them (apart from experimentation).
65 // See also:
66 // http://stackoverflow.com/questions/12332772/why-arent-container-move-assignment-operators-noexcept
67 // NOTE: now we do not use the allocator at all, as we cannot guarantee it has a standard layout. Keep the comments
68 // above as an historical reference, might come useful in the future :)

71 {
72  PIRANHA_TT_CHECK(is_container_element,T);

79  // NOTE: this bit of TMP is to avoid checking an always-false condition on reserve() on most platforms, which triggers a compiler
80  // warning on GCC 4.7.
81  static const std::size_t max_alloc_size = std::numeric_limits<std::size_t>::max() / sizeof(value_type);
82  static const bool need_reserve_check = std::numeric_limits<size_type>::max() > max_alloc_size;
83  static bool reserve_check_size(const size_type &, const std::false_type &)
84  {
85  return false;
86  }

88  {

94  using const_iterator = const_pointer;
95  dynamic_storage() : m_tag(0u),m_size(0u),m_capacity(0u),m_ptr(nullptr) {}
96  dynamic_storage(dynamic_storage &&other) noexcept : m_tag(0u),m_size(other.m_size),m_capacity(other.m_capacity),m_ptr(other.m_ptr)
97  {
98  // Erase the other.
99  other.m_size = 0u;
100  other.m_capacity = 0u;
101  other.m_ptr = nullptr;
102  }
103  dynamic_storage(const dynamic_storage &other) :
104  m_tag(0u),m_size(0u),m_capacity(other.m_size), // NOTE: when copying, we set the capacity to the same value of the size.
105  m_ptr(nullptr)
106  {
107  // Obtain storage. Will just return nullptr if requested size is zero.
108  m_ptr = allocate(other.m_size);
109  // Attempt to copy-construct the elements from other.
110  try {
111  for (; m_size < other.m_size; ++m_size) {
112  construct(m_ptr + m_size,other[m_size]);
113  }
114  } catch (...) {
115  // Roll back the construction and deallocate before re-throwing.
116  destroy_and_deallocate();
117  throw;
118  }
119  }
120  ~dynamic_storage()
121  {
122  // NOTE: here we should replace with bidirectional tt, if we ever implement it.
123  PIRANHA_TT_CHECK(is_forward_iterator,iterator);
124  PIRANHA_TT_CHECK(is_forward_iterator,const_iterator);

196  void reserve(const size_type &new_capacity)
197  {
198  piranha_assert(consistency_checks());
199  // No need to do anything if we already have enough capacity.
200  if (new_capacity <= m_capacity) {
201  return;
202  }
203  // Check that we are not asking too much.
204  // NOTE: the first check is redundant right now, just keep it around in case max_size changes in the future.
205  if (unlikely(/*new_capacity > max_size ||*/ reserve_check_size(new_capacity,std::integral_constant<bool,need_reserve_check>()))) {
206  piranha_throw(std::bad_alloc,);
207  }
208  // Start by allocating the new storage. New capacity is at least one at this point.
209  pointer new_storage = allocate(new_capacity);
210  assert(new_storage != nullptr);
211  // Move in existing elements. Consistency checks ensure
212  // that m_size is not greater than m_capacity and, by extension, new_capacity.
213  for (size_type i = 0u; i < m_size; ++i) {
214  construct(new_storage + i,std::move((*this)[i]));
215  }
216  // Destroy and deallocate original content.
217  destroy_and_deallocate();
218  // Move in the new pointer and capacity.
219  m_capacity = new_capacity;
220  m_ptr = new_storage;
221  }
222  size_type capacity() const
223  {
224  return m_capacity;
225  }
226  std::size_t hash() const
227  {
228  return detail::vector_hasher(*this);
229  }
230  void resize(const size_type &new_size)
231  {
232  // NOTE: another redundant size check at the moment.
233  // if (unlikely(new_size > max_size)) {
234  // piranha_throw(std::bad_alloc,);
235  // }
236  if (new_size == m_size) {
237  return;
238  }
239  // The storage we are going to operate on is either the old one, if it has enough capacity,
240  // or new storage.
241  const bool new_storage = (m_capacity < new_size);
242  pointer storage = new_storage ? allocate(new_size) : m_ptr;
243  // NOTE: storage cannot be nullptr:
244  // - if new storage, new_size has to be at least 1 (new_size > m_capacity);
245  // - if not new storage, new_size <= m_capacity; m_ptr can be null only if capacity is 0, but then
246  // m_size is zero and new_size is 0 as well, and the function never arrived here because of the check above.
247  piranha_assert(storage != nullptr);
248  // Default-construct excess elements. We need to do this regardless of where the storage is coming from.
249  // This is also the only place we care about exception handling.
250  size_type i = m_size;
251  try {
252  for (; i < new_size; ++i) {
253  construct(storage + i);
254  }
255  } catch (...) {
256  // Roll back and dealloc.
257  for (size_type j = m_size; j < i; ++j) {
258  destroy(storage + j);
259  }
260  deallocate(storage);
261  throw;
262  }
263  // NOTE: no more exceptions thrown after this point.
264  if (new_storage) {
265  // Move in old elements into the new storage. As we had to increase the capacity,
266  // we know that new_size has to be greater than the old one, hence all old elements
267  // need to be moved over.
268  for (size_type i = 0u; i < m_size; ++i) {
269  construct(storage + i,std::move((*this)[i]));
270  }
271  // Erase the old content and assign new.
272  destroy_and_deallocate();
273  m_capacity = new_size;
274  m_ptr = storage;
275  } else {
276  // Destroy excess elements in the old storage.
277  for (size_type i = new_size; i < m_size; ++i) {
278  destroy(storage + i);
279  }
280  }
281  // In any case, we need to update the size.
282  m_size = new_size;
283  }
284  private:
285  template <typename ... Args>
286  static void construct(pointer p, Args && ... args)
287  {
288  ::new (static_cast<void *>(p)) value_type(std::forward<Args>(args)...);
289  }
290  static void destroy(pointer p)
291  {
292  p->~value_type();
293  }
294  // Obtain new storage, and throw an error in case something goes wrong.
295  // NOTE: no need to check for zero, will aready return nullptr in that case.
296  static pointer allocate(const size_type &s)
297  {
298  return static_cast<pointer>(aligned_palloc(0u,static_cast<std::size_t>(s * sizeof(value_type))));
299  }
300  static void deallocate(pointer p)
301  {
302  aligned_pfree(0u,p);
303  }
304  // Common implementation of push_back().
305  template <typename U>
306  void push_back_impl(U &&x)
307  {
308  piranha_assert(consistency_checks());
309  if (unlikely(m_capacity == m_size)) {
310  increase_capacity();
311  }
312  construct(m_ptr + m_size,std::forward<U>(x));
313  ++m_size;
314  }
315  void destroy_and_deallocate()
316  {
317  piranha_assert(consistency_checks());
318  for (size_type i = 0u; i < m_size; ++i) {
319  // NOTE: could use POD optimisations here in principle.
320  destroy(m_ptr + i);
321  }
322  // NOTE: no need to check for nullptr, aligned_pfree already does it.
323  deallocate(m_ptr);
324  }
325  // Will try to double the capacity, or, in case this is not possible,
326  // will set the capacity to max_size. If the initial capacity is already max,
327  // then will throw an error.
328  void increase_capacity()
329  {
330  if (unlikely(m_capacity == max_size)) {
331  piranha_throw(std::bad_alloc,);
332  }
333  // NOTE: capacity should double, but without going past max_size, and in case it is zero it should go to 1.
334  const size_type new_capacity = (m_capacity > max_size / 2u) ? max_size : ((m_capacity != 0u) ?
335  static_cast<size_type>(m_capacity * 2u) : static_cast<size_type>(1u));
336  reserve(new_capacity);
337  }
338  bool consistency_checks()
339  {
340  // Size cannot be greater than capacity.
341  if (unlikely(m_size > m_capacity)) {
342  return false;
343  }
344  // If we allocated something, capacity cannot be zero.
345  if (unlikely(m_ptr != nullptr && m_capacity == 0u)) {
346  return false;
347  }
348  // If we have nothing allocated, capacity must be zero.
349  if (unlikely(m_ptr == nullptr && m_capacity != 0u)) {
350  return false;
351  }
352  return true;
353  }
354  private:
355  unsigned char m_tag;
356  size_type m_size;
357  size_type m_capacity;
358  pointer m_ptr;

395 // NOTE: here we are using a special rule from 9.2. If the union is standard-layout, and it contains two or more
396 // standard-layout classes, it will be possible to inspect the value of the common initial part of any of them. That is,
397 // we can access the initial m_tag member of the static and dynamic storage via both st and dy, and use it to detect
398 // which class of the union is active.
399 // See also:
400 // http://stackoverflow.com/questions/18564497/writing-into-the-last-byte-of-a-class
401 // http://www.informit.com/guides/content.aspx?g=cplusplus&seqNum=556
402 // http://en.wikipedia.org/wiki/C%2B%2B11#Unrestricted_unions
403 template <typename T, typename S>
404 union small_vector_union
405 {
406  using s_storage = typename std::conditional<S::value == 0u,static_vector<T,auto_static_size<T>::value>,static_vector<T,S::value>>::type;
407  using d_storage = dynamic_storage<T>;
408  // NOTE: each constructor must be invoked explicitly.
409  small_vector_union():m_st() {}
410  small_vector_union(const small_vector_union &other)
411  {
412  if (other.is_static()) {
413  ::new (static_cast<void *>(&m_st)) s_storage(other.g_st());
414  } else {
415  ::new (static_cast<void *>(&m_dy)) d_storage(other.g_dy());
416  }
417  }
418  small_vector_union(small_vector_union &&other) noexcept
419  {
420  if (other.is_static()) {
421  ::new (static_cast<void *>(&m_st)) s_storage(std::move(other.g_st()));
422  } else {
423  ::new (static_cast<void *>(&m_dy)) d_storage(std::move(other.g_dy()));
424  }
425  }
426  ~small_vector_union()
427  {
428  PIRANHA_TT_CHECK(std::is_standard_layout,s_storage);
429  PIRANHA_TT_CHECK(std::is_standard_layout,d_storage);
430  PIRANHA_TT_CHECK(std::is_standard_layout,small_vector_union);
431  if (is_static()) {
432  g_st().~s_storage();
433  } else {
434  g_dy().~d_storage();
435  }
436  }
437  small_vector_union &operator=(small_vector_union &&other) noexcept
438  {
439  if (unlikely(this == &other)) {
440  return *this;
441  }
442  if (is_static() == other.is_static()) {
443  if (is_static()) {
444  g_st() = std::move(other.g_st());
445  } else {
446  g_dy() = std::move(other.g_dy());
447  }
448  } else {
449  if (is_static()) {
450  // static vs dynamic.
451  // Destroy static.
452  g_st().~s_storage();
453  // Move construct dynamic from other.
454  ::new (static_cast<void *>(&m_dy)) d_storage(std::move(other.g_dy()));
455  } else {
456  // dynamic vs static.
457  g_dy().~d_storage();
458  ::new (static_cast<void *>(&m_st)) s_storage(std::move(other.g_st()));
459  }
460  }
461  return *this;
462  }
463  small_vector_union &operator=(const small_vector_union &other)
464  {
465  if (likely(this != &other)) {
466  *this = small_vector_union(other);
467  }
468  return *this;
469  }
470  bool is_static() const
471  {
472  return static_cast<bool>(m_st.m_tag);
473  }
474  // Getters.
475  const s_storage &g_st() const
476  {
477  piranha_assert(is_static());
478  return m_st;
479  }
480  s_storage &g_st()
481  {
482  piranha_assert(is_static());
483  return m_st;
484  }
485  const d_storage &g_dy() const
486  {
487  piranha_assert(!is_static());
488  return m_dy;
489  }
490  d_storage &g_dy()
491  {
492  piranha_assert(!is_static());
493  return m_dy;
494  }
495  s_storage m_st;
496  d_storage m_dy;
497 };

527 // - the m_size member of dynamic and static could be made a signed integer, the sign establishing the storage type
528 // and drop the m_tag member. This in principle would allow to squeeze some extra space from the static vector
529 // but not sure this is worth it;

531 // - in the dynamic storage, it looks like on 64bit we can bump up the size member to 16 bit without changing size,

533 template <typename T, typename S = std::integral_constant<std::size_t,0u>>
534 class small_vector
535 {
536  PIRANHA_TT_CHECK(is_container_element,T);

542  using size_type_impl = max_int<typename s_storage::size_type,typename d_storage::size_type>;
543  template <typename U>
544  static constexpr U get_max(const U &a, const U &b)
545  {
546  return (a > b) ? a : b;
547  }

554  using size_type = size_type_impl;

557  static const bool need_resize_check = std::numeric_limits<size_type>::max() > d_storage::max_size;
558  static bool resize_check_size(const size_type &, const std::false_type &)
559  {

570  using value_type = T;

574  using const_iterator = value_type const *;
575  private:
576  // NOTE: here we should replace with bidirectional tt, if we ever implement it.

655  template <typename U, init_list_enabler<U> = 0>
656  explicit small_vector(std::initializer_list<U> l)
657  {
658  // NOTE: push_back has strong exception safety.
659  for (const U &x : l) {

879  void resize(const size_type &size)
880  {
881  if (m_union.is_static()) {
882  if (size <= max_static_size) {
883  m_union.g_st().resize(static_cast<typename s_storage::size_type>(size));
884  } else {
885  // NOTE: this check is a repetition of an existing check in dynamic storage's
886  // resize. The reason for putting it here as well is to ensure the safety
887  // of the static_cast below.
888  if (unlikely(resize_check_size(size,std::integral_constant<bool,need_resize_check>()))) {
889  piranha_throw(std::bad_alloc,);
890  }
891  // Move the existing elements into new dynamic storage.
892  const auto d_size = static_cast<typename d_storage::size_type>(size);
893  d_storage tmp_d;
894  tmp_d.reserve(d_size);
895  // NOTE: this will not throw, as tmp_d is guaranteed to be of adequate size
896  // and thus push_back() will not fail.
897  std::move(m_union.g_st().begin(),m_union.g_st().end(),std::back_inserter(tmp_d));
898  // Fill in the missing elements.
899  tmp_d.resize(d_size);
900  // Destroy static, move in dynamic.
901  m_union.g_st().~s_storage();
902  ::new (static_cast<void *>(&(m_union.m_dy))) d_storage(std::move(tmp_d));
903  }
904  } else {
905  if (unlikely(resize_check_size(size,std::integral_constant<bool,need_resize_check>()))) {
906  piranha_throw(std::bad_alloc,);
907  }
908  m_union.g_dy().resize(static_cast<typename d_storage::size_type>(size));
909  }

947  }

951  // to short int on return.
952  template <typename U>
953  struct adder<U,typename std::enable_if<std::is_integral<U>::value>::type>

960  template <typename U>
961  void push_back_impl(U &&x)
962  {
963  if (m_union.is_static()) {
964  if (m_union.g_st().size() == m_union.g_st().max_size) {
965  // Create a new dynamic vector, and move in the current
966  // elements from static storage.
967  using d_size_type = typename d_storage::size_type;
968  // NOTE: this check ensures that the d_storage::max_size is strictly greater than
969  // the current (static) size (equal to max static size). This means we can compute safely
970  // current size + 1 while casting to dynamic storage size type and try to allocate enough space for
971  // the std::move and push_back() below.
972  // NOTE: this check is analogous to current_size + 1 > d_storage::max_size, i.e., the same
973  // check in resize(), but it will use static const variables written as this.
974  if (unlikely(s_storage::max_size >= d_storage::max_size)) {
975  piranha_throw(std::bad_alloc,);
976  }
977  d_storage tmp_d;
978  tmp_d.reserve(static_cast<d_size_type>(static_cast<d_size_type>(m_union.g_st().max_size) + 1u));
979  std::move(m_union.g_st().begin(),m_union.g_st().end(),std::back_inserter(tmp_d));
980  // Push back the new element.
981  tmp_d.push_back(std::forward<U>(x));
982  // Now destroy the current static storage, and move-construct new dynamic storage.
983  // NOTE: in face of custom allocators here we should be ok, as move construction
984  // of custom alloc will not throw and it will preserve the ownership of the moved-in elements.
985  m_union.g_st().~s_storage();
986  ::new (static_cast<void *>(&(m_union.m_dy))) d_storage(std::move(tmp_d));
987  } else {
988  m_union.g_st().push_back(std::forward<U>(x));
989  }
990  } else {
991  // In case we are already in dynamic storage, don't do anything special.
992  m_union.g_dy().push_back(std::forward<U>(x));
993  }
994  }

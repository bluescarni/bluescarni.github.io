792 // Detect the availability of std::iterator_traits on type It, plus a couple more requisites from the
793 // iterator concept.
794 // NOTE: this needs also the is_swappable type trait, but this seems to be difficult to implement in C++11. Mostly because
795 // it seems that:
796 // - we cannot detect a specialised std::swap (so if it is not specialised, it will pick the default implementation
797 // which could fail in the implementation without giving hints in the prototype),
798 // - it's tricky to fulfill the requirement that swap has to be called unqualified (cannot use 'using std::swap' within a decltype()
799 // SFINAE, might be doable with automatic return type deduction for regular functions in C++14?).
800 template <typename It>
801 struct has_iterator_traits
802 {
803  using it_tags = std::tuple<std::input_iterator_tag,std::output_iterator_tag,std::forward_iterator_tag,
804  std::bidirectional_iterator_tag,std::random_access_iterator_tag>;
805  PIRANHA_DECLARE_HAS_TYPEDEF(difference_type);
